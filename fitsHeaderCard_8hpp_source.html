409 inline int fitsHeaderCard::write(fitsfile *fptr)
410 {
411 // int fstatus;
412  switch(type)
413  {
414  case TSTRING:
415  {
416  return fits_write_key<char *>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
417  }
418  case TBYTE:
419  {
420  unsigned char b = Int();
421  return fits_write_key<unsigned char>(fptr, (char *) keyword.c_str(), &b, (char *) comment.c_str());
422  }
423  case TSBYTE:
424  {
425  signed char sc = Int();
426  return fits_write_key<signed char>(fptr, (char *) keyword.c_str(), &sc, (char *) comment.c_str());
427  }
428  case TSHORT:
429  {
430  short s = Int();
431  return fits_write_key<short>(fptr, (char *) keyword.c_str(), &s, (char *) comment.c_str());
432  }
433  case TUSHORT:
434  {
435  unsigned short us = Int();
436  return fits_write_key<unsigned short>(fptr, (char *) keyword.c_str(), &us, (char *) comment.c_str());
437  }
438  case TINT:
439  {
440  int i = Int();
441  return fits_write_key<int>(fptr, (char *) keyword.c_str(), &i, (char *) comment.c_str());
442  }
443  case TLONG:
444  {
445  long l = Long();
446  return fits_write_key<long>(fptr, (char *) keyword.c_str(), &l, (char *) comment.c_str());
447  }
448  case TLONGLONG:
449  {
450  long long ll = longLong();
451  return fits_write_key<long long>(fptr, (char *) keyword.c_str(), &ll, (char *) comment.c_str());
452  }
453  case TULONG:
454  {
455  unsigned long ul = unsignedLong();
456  return fits_write_key<unsigned long>(fptr, (char *) keyword.c_str(), &ul, (char *) comment.c_str());
457  }
458  case TFLOAT:
459  {
460  float f = Float();
461  return fits_write_key<float>(fptr, (char *) keyword.c_str(), &f, (char *) comment.c_str());
462  }
463  case TDOUBLE:
464  {
465  double d = Double();
466  return fits_write_key<double>(fptr, (char *) keyword.c_str(), &d, (char *) comment.c_str());
467  }
468  case fitsTCOMMENT:
469  {
470  return fits_write_comment(fptr, (char *) comment.c_str());
471  }
472  case fitsTHISTORY:
473  {
474  return fits_write_history(fptr, (char *) comment.c_str());
475  }
476  default:
477  {
478  return fits_write_key<fitsUnknownType>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
479  }
480  }
