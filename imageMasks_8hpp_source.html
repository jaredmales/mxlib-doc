209  )

215 
216  if(max_q == 0) max_q = 360.;
217 
218  size_t msize = ((math::pi<double>()*(max_r*max_r - min_r*min_r)) * (max_q-min_q)/360.) *1.01 + 1;
219 
220  //This was tested, this is slightly faster than resize with an erase.
221  idx.reserve(msize);
222 
223  int x0 = xcen+min_x;
224  if(x0 < 0) x0 = 0;
225  int x1 = xcen+max_x+1;
226  if(x1 > rIm.rows()) x1 = rIm.rows();
227  int y0 = ycen+min_y;
228  if(y0 < 0) y0 = 0;
229  int y1 = ycen+max_y+1;
230  if(y1 > rIm.cols()) y1 = rIm.cols();
231 
232  for(int i = x0; i< x1; ++i)
233  {
234  for(int j = y0; j< y1; ++j)
235  {
236  if(rIm(i,j) >= min_r && rIm(i,j) < max_r && qIm(i,j) >= min_q && qIm(i,j) < max_q)
237  {
238  if( mask )
239  {
240  if( (*mask)(i,j) == 0) continue;
241  }
242 
243  idx.push_back(j*rIm.cols() + i);
244  }
245  }
246  }

315 /** The circle is describe by its center coordinates and radius. Any value can be set for the mask.

317  *
318  * \tparam arrayT is an Eigen-like type.
319  *
320  * \ingroup image_masks
321  */
322 template<class arrayT>
323 void maskCircle( arrayT & m, ///< [in/out] the image to be masked, is modified.
324  typename arrayT::Scalar xcen, ///< [in] the x coordinate of the center of the circle
325  typename arrayT::Scalar ycen, ///< [in] the y coordinate of the center of the circle
326  typename arrayT::Scalar rad, ///< [in] the radius of the circle
327  typename arrayT::Scalar val, ///< [in] the mask value.
328  typename arrayT::Scalar pixbuf = 0.5 ///< [in] [optional] buffer for radius comparison. Default is 0.5 pixels.
329  )
330 {
331  size_t l0 = m.rows();
332  size_t l1 = m.cols();
333 
334  typename arrayT::Scalar rr;
335 
336 
337  for(size_t c=0; c < m.cols(); c++)
338  {
339  for(size_t r=0; r < m.rows(); r++)
340  {
341  rr = sqrt( pow(r-xcen, 2) + pow(c-ycen, 2) );
342 
343  if(rr <= rad+pixbuf) m(r,c) = val;
344  }

366 ///Mask an ellipse in an image.
367 /** The ellipse is describe by its center coordinates and x and y direction radii (the semi-major and -minor axes, in either order). Any value can be set for the mask,
368  * with 0 being the default. The ellipse axes must be horizontal and vertical, i.e. there is no rotation provided.
369  *
370  * \tparam arrayT is an Eigen-like type.
371  *
372  * \ingroup image_masks
373  */
374 template<class arrayT>
375 void maskEllipse( arrayT & m, ///< [in/out] the image to be masked, is modified.
376  typename arrayT::Scalar xcen, ///< [in] the x coordinate of the center of the ellipse
377  typename arrayT::Scalar ycen, ///< [in] the y coordinate of the center of the ellipse
378  typename arrayT::Scalar xrad, ///< [in] the x radius of the ellipse
379  typename arrayT::Scalar yrad, ///< [in] the y radius of the ellipse
380  typename arrayT::Scalar val = 0, ///< [in] [optional] the mask value. Default is 0.
381  typename arrayT::Scalar pixbuf = 0.5 ///< [in] [optional] buffer for radius comparison. Default is 0.5 pixels.
382  )
383 {
384  size_t l0 = m.rows();
385  size_t l1 = m.cols();
386 
387  typename arrayT::Scalar r;
388  typename arrayT::Scalar x;
389  typename arrayT::Scalar y;
390  typename arrayT::Scalar xe, ye;
391  typename arrayT::Scalar rad;
392 
393 
394  for(size_t i=0; i < l0; i++)
395  {
396  x = i-xcen;
397  for(size_t j=0; j < l1; j++)
398  {
399  y = j-ycen;
400 
401  xe = (pow(xrad*yrad,2)/ (pow(yrad,2) + pow(xrad*y/x,2)));
402  ye = (pow(yrad,2) - xe*pow(yrad/xrad,2));
403 
404  rad = sqrt(xe + ye);
405 
406  r = sqrt( pow(i-xcen, 2) + pow(j-ycen, 2) );
407 
408  if(r <= rad+pixbuf) m(i,j) = val;
409  }
410  }
