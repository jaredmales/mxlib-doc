88  int plan( const std::vector<realT> & freq, ///< [in] The Frequency grid of the input PSD
89  const std::vector<realT> & fmPSD, ///< [in] The Fourier mode PSD.
90  const std::vector<std::complex<realT>> & fmXferFxn, ///< [in] The complex error transfer function, as a function of freq
91  const std::vector<realT> & nPSD, ///< [in] The open-loop noise PSD
92  const std::vector<std::complex<realT>> & nXferFxn ///< [in] The noise transfer function, as a function of freq
93  )
94  {
95 
96  //Calculate the Fourier Mode variance for normalization
97  m_fmVar = sigproc::psdVar( freq, fmPSD);
98  //and the noise variance
99  m_nVar = sigproc::psdVar( freq, nPSD);
100 
101  if(freq[0] == 0) m_hasZero = true;
102 
103 
104  //First augment to two-sided DFT form
105  mx::sigproc::augment1SidedPSD( m_psd2, fmPSD, !m_hasZero, 0.5);
106  mx::sigproc::augment1SidedPSD( m_npsd2, nPSD, !m_hasZero, 0.5);
107 
108  //And augment the xfer fxns to two sided form
109  sigproc::augment1SidedPSD( m_xfer2, fmXferFxn, !m_hasZero, 1.0);
110  sigproc::augment1SidedPSD( m_nxfer2, nXferFxn, !m_hasZero, 1.0);
111 
112  m_filt.psd(m_psd2,freq[1]-freq[0]);
113  m_nfilt.psd(m_npsd2,freq[1]-freq[0]);
114 
115  m_fft.plan(m_psd2.size());
116  m_fftB.plan(m_psd2.size(), MXFFT_BACKWARD);
117 
118  m_tform1.resize(m_psd2.size());
119  m_tform2.resize(m_psd2.size());
120 
121  m_Ntform1.resize(m_psd2.size());
122  m_Ntform2.resize(m_psd2.size());
123 
124  //The two modal amplitude series
125  m_fm_n.resize(m_psd2.size());
126  m_fm_nm.resize(m_psd2.size());
127 
128  //The two noise amplitude series
129  m_N_n.resize(m_psd2.size());
130  m_N_nm.resize(m_psd2.size());
131 
132  //The time sampling
133  realT m_dt = 1./(2*freq.back());
134 
135  //Indices for getting the middle half
136  m_Nwd = 1.0*m_psd2.size();
137  m_NwdStart = 0.5*m_psd2.size() - 0.5*m_Nwd;
138 
139  m_Nsamp = 1.0*m_psd2.size();
140  m_NsampStart = 0.5*m_Nwd - 0.5*m_Nsamp;
141 
142  //The speckle amplitude
143  m_vnl.resize( m_Nwd );
144 
145  return 0;

150 /// Calculate the PSD of the speckle intensity given the PSD of Fourier mode amplitude
151 /**
152  * Does so by generating a time-series from the PSD using Fourier-domain convolution, and calculating
153  * the average periodogram.
154  *
155  * A Hann window is used.
156  *
157  * Statistics are not currently normalized. You need to normalize to match expected contrast if desired.
158  *
159  * Will also calculate the binned-variances in the generated time-series, if the arguments vars and bins are not null pointers.
160  *
161  * \todo Figure out what's going on with psdFilter normalization!
162  * \todo probably need to not do overlapped averaging in periodogram. Could drop mean-sub then.
163  */
164 int singleTS( std::vector<realT> & spTS /**< [out] The speckle amplitude time-series. */)
165 {
166  //Generate the time-series
167  for(int i=0; i< m_psd2.size(); ++i)
168  {
169  m_fm_n[i] = m_normVar;
170  m_N_n[i] = m_normVar;
171  m_N_nm[i] = m_normVar;
172  }
173 
174  //Filter and normalize the fourier mode time series
175  m_filt(m_fm_n);
176  math::vectorMeanSub(m_fm_n);
177  realT actvar = math::vectorVariance(m_fm_n);
178 
179  realT norm = sqrt(m_fmVar/actvar);
180 
181  for(size_t q=0; q<m_fm_n.size(); ++q) m_fm_n[q] *= norm;
182 
183  //And move it to the Fourier domain
184  m_fft(m_tform1.data(), m_fm_n.data());
185 
186  //Filter and normalize the measurement mode time series
187  m_nfilt.filter(m_N_n);
188  m_nfilt.filter(m_N_nm);
189 
190  realT Nactvar = 0.5*(math::vectorVariance(m_N_n) + math::vectorVariance(m_N_nm));
191  norm = sqrt(m_nVar/Nactvar);
192  for(size_t q=0; q<m_fm_n.size(); ++q) m_N_n[q] *= norm;
193  for(size_t q=0; q<m_fm_n.size(); ++q) m_N_nm[q] *= norm;
194 
195  //And move them to the Fourier domain
196  m_fft(m_Ntform1.data(), m_N_n.data());
197  m_fft(m_Ntform2.data(), m_N_nm.data());
198 
199  std::complex<realT> scale = std::complex<realT>((m_tform1.size()),0) ;
200 
201  //Apply the modal phase shift, and apply the measurement noise.
202  for(size_t m=0;m<m_tform1.size();++m)
203  {
204  // Apply the phase shift to form the 2nd time series
205  m_tform2[m] = m_tform1[m]*exp( std::complex<realT>(0, math::half_pi<realT>() ));
206 
207  //Apply the augmented ETF to two time-series
208  m_tform1[m] *= m_xfer2[m]/scale;
209  m_tform2[m] *= m_xfer2[m]/scale;
210 
211  //Ntform2[m] = Ntform1[m]*exp( std::complex<realT>(0, math::half_pi<realT>() ));
212 
213  m_Ntform1[m] *= m_nxfer2[m]/scale;
214  m_Ntform2[m] *= m_nxfer2[m]/scale;
215  }
216 
217  //<<<<<<<<****** Transform back to the time domain.
218  m_fftB(m_fm_n.data(), m_tform1.data());
219  m_fftB(m_fm_nm.data(), m_tform2.data());
220  m_fftB(m_N_n.data(), m_Ntform1.data());
221  m_fftB(m_N_nm.data(), m_Ntform2.data());
222 
223  //Calculate the speckle amplitude and mean-subtract
224  for(int i= 0; i< m_Nwd; ++i)
225  {
226  realT h1 = m_fm_n[i+m_NwdStart]+m_N_n[i+m_NwdStart];
227  realT h2 = m_fm_nm[i+m_NwdStart]+m_N_nm[i+m_NwdStart];
228 
229  m_vnl[i] = (pow(h1,2) + pow(h2,2));
230  }
231 
232  //Extract middle sample and mean subtract
233  spTS.resize(m_Nsamp);
234  for(int i=0; i<m_Nsamp; ++i) spTS[i] = m_vnl[i+m_NsampStart];
235 
236 
237  return 0;
238 }
