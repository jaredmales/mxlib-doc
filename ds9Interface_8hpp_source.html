362 inline
363 int ds9Interface::connect()
364 {
365 
366  if(xpa)
367  {
368  shutdown();
369  xpa = NULL;
370  }
371 
372  if(xpa == NULL)
373  {
374  xpa = XPAOpen(NULL);
375  }
376 
377  int n = 1;
378  char *names[1];
379  names[0] = NULL;
380 
381  std::string tmpl;
382  if( m_title.find(':', 0) == std::string::npos)
383  {
384  if(m_title == "") m_title = "ds9";
385 
386  tmpl = "DS9:";
387  tmpl += m_title;
388  }
389  else
390  {
391  tmpl = m_title;
392  }
393 
394  char paramlist[] = "gsi";
395 
396  int rv = XPAAccess(xpa, const_cast<char *>(tmpl.c_str()), paramlist, NULL, names, NULL, n);
397 
398  if(rv == 0)
399  {
400  if( spawn() != 0) return -1;
401  if(names[0])
402  {
403  free(names[0]);
404  names[0] = NULL;
405  }
406 
407  int slept = 0;
408  while(rv == 0 && slept < DS9INTERFACE_SPAWN_TIMEOUT)
409  {
410  rv = XPAAccess(xpa, const_cast<char *>(tmpl.c_str()), paramlist, NULL, names, NULL, n);
411 
412  usleep(DS9INTERFACE_SPAWN_SLEEP*1000);
413  slept += DS9INTERFACE_SPAWN_SLEEP;
414  }
415 
416  if(rv == 0)
417  {
418  std::cerr << "ds9Interface: failed to connect after attempting to spawn. Timed out.\n";
419  if(names[0]) free(names[0]);
420  return -1;
421  }
422  }
423 
424  char * st = strchr(names[0], ' ');
425 
426  m_ipAndPort = st + 1;
427 
428  free(names[0]);
429 
430  m_connected = true;
431 
432  return 0;
433 }

436 int ds9Interface::spawn()
437 {
438  if(m_title == "" || m_title == "*") m_title = "ds9";
439 
440  int pid = fork();
441  if (pid==0)
442  {
443  /* Create a new SID for the child process so it is detached*/
444  pid_t sid = setsid();
445 
446  if (sid < 0)
447  {
448  std::cerr << "ds9Interface: failed to detach.\n";
449  perror("ds9Interface");
450  }
451 
452  errno = 0;
453  execlp("ds9", "ds9", "-title", m_title.c_str(), (char *) 0);
454 
455  std::cerr << "ds9Interface: spawning failed, execlp returned.\n";
456  perror("ds9Interface");
457 
458  return -1;
459  }
460 
461  return 0;

605 int ds9Interface::display( const void * im,
606  int bitpix,
607  size_t pixsz,
608  size_t dim1,
609  size_t dim2,
610  size_t dim3,
611  int frame
612  )
613 {
614  size_t tot_size;
615  char cmd[DS9INTERFACE_CMD_MAX_LENGTH];
616 
617  if(frame < 1)
618  {
619  std::cerr << "ds9Interface: frame must >= 1\n" << "\n";
620  return -1;
621  }
622 
623  if(!m_connected) if(connect() < 0) return -1;
624 
625  if(addframe(frame) < 0)
626  {
627  m_connected = false;
628  return -1;
629  }
630  //Calculate total size
631  tot_size= pixsz;
632  tot_size*=dim1;
633  tot_size*=dim2;
634  tot_size*=dim3;
635 
636  bool realloc = false;
637 
638  //Re-allocate shared memory if necessary
639  if(tot_size > m_segs[frame-1].size)
640  {
641  if( m_segs[frame-1].size > 0 )
642  {
643  m_segs[frame-1].detach();
644  }
645  m_segs[frame-1].create(tot_size);
646 
647  realloc = true;
648  }
649  else
650  {
651  if( dim1 != m_segs[frame-1].dim1 || dim2 != m_segs[frame-1].dim2 || dim3 != m_segs[frame-1].dim3 || bitpix != m_segs[frame-1].bitpix)
652  {
653  realloc = true; //force a new shm command
654  }
655  }
656 
657  memcpy( m_segs[frame-1].addr, im, tot_size );
658 
659  m_segs[frame-1].dim1 = dim1;
660  m_segs[frame-1].dim2 = dim2;
661  m_segs[frame-1].dim3 = dim3;
662  m_segs[frame-1].bitpix = bitpix;
663 
664  if(realloc)
665  {
666  //Handle single image so that the cube dialog doesn't open up if dim3=1
667  if(dim3 == 1)
668  {
669  snprintf(cmd, DS9INTERFACE_CMD_MAX_LENGTH, "shm array shmid %i [xdim=%zu,ydim=%zu,bitpix=%i]",
670  m_segs[frame-1].shmemid,
671  dim1, dim2, bitpix);
672  }
673  else
674  {
675  snprintf(cmd, DS9INTERFACE_CMD_MAX_LENGTH, "shm array shmid %i [xdim=%zu,ydim=%zu,zdim=%zu,bitpix=%i]",
676  m_segs[frame-1].shmemid,
677  dim1, dim2, dim3, bitpix);
678  }
679  }
680  else
681  {
682  snprintf(cmd, DS9INTERFACE_CMD_MAX_LENGTH, "update");
683  }
684 
685  int rv = XPASet(cmd);
686 
687  if(rv != 0)
688  {
689  std::cerr << "ds9Interface: sending shm array command to ds9 failed.\n";
690  m_connected = false;
691  return -1;
692  }
693 
694  if( m_regionsPreserved != m_preserveRegions ) togglePreserveRegions(m_preserveRegions);
695  if( m_panPreserved != m_preservePan ) togglePreservePan(m_preservePan);
696 
697 
698  return 0;
699 
700 }
