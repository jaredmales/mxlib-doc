41/// A smart-pointer wrapper for cuda device pointers.

43 * \ingroup cuda

46struct cudaPtr

49 // typedef typename cudaType<T>::hostType hostPtrT;

182template <typename T>
183int cudaPtr<T>::resize( size_t sz )
184{
185 if( m_size == sz )
186 return 0;
187
188 m_size = sz;
189
190 cudaError_t rv = cudaMalloc( (void **)&m_devicePtr, sz * sizeof( devicePtrT ) );
191
192 if( rv != cudaSuccess )
193 {
194 std::cerr << "Error from cudaMalloc: ";
195 printf( "[%s] %s\n", cudaGetErrorName( rv ), cudaGetErrorString( rv ) );
196 }
197
198 return 0;
199}

219template <typename T>
220int cudaPtr<T>::free()
221{
222 if( m_devicePtr )
223 {
224 int rv = cudaFree( m_devicePtr );
225
226 if( rv != cudaSuccess )
227 {
228 std::cerr << "Cuda Free Error \n";
229 return rv;
230 }
231 }
232
233 m_devicePtr = 0;
234 m_size = 0;
235
236 return 0;
237}

239template <typename T>
240int cudaPtr<T>::upload( const hostPtrT *src )
241{
242 // Copy host memory to device
243 int rv = cudaMemcpy( m_devicePtr, src, m_size * sizeof( devicePtrT ), cudaMemcpyHostToDevice );
244
245 if( rv != cudaSuccess )
246 {
247 std::cerr << "Cuda Memcpy error \n";
248 return rv;
249 }
250
251 return 0;
252}

254template <typename T>
255int cudaPtr<T>::upload( const hostPtrT *src, size_t sz )
256{
257 int rv;
258
259 rv = resize( sz );
260
261 if( rv )
262 return rv;
263
264 return upload( src );
265}

272
