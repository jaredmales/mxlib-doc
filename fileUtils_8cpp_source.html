88std::vector<std::string> getFileNames( const std::string &directory,
89 const std::string &prefix,
90 const std::string &substr,
91 const std::string &extension )
92{
93 std::vector<std::string> vect;
94 if( std::filesystem::exists( directory ) )
95 {
96 if( std::filesystem::is_directory( directory ) )
97 {
98 try
99 {
100 bool hasext = false;
101 std::string _ext;
102 if( extension.size() > 0 )
103 {
104 if( extension[0] != '.' )
105 {
106 _ext = '.';
107 }
108
109 _ext += extension;
110
111 hasext = true;
112 }
113
114 bool hasprefix = false;
115 if( prefix != "" )
116 {
117 hasprefix = true;
118 }
119
120 bool hassub = false;
121 if( substr != "" )
122 {
123 hassub = true;
124 }
125
126 std::filesystem::directory_iterator it{ directory };
127 auto it_end = std::filesystem::directory_iterator{};
128 for( it; it != it_end; ++it )
129 {
130 if( hasext )
131 {
132 if( it->path().extension() != _ext )
133 {
134 continue;
135 }
136 }
137
138 std::string p = it->path().filename().generic_string();
139
140 if( hasprefix )
141 {
142 if( p.size() < prefix.size() )
143 {
144 continue;
145 }
146 else
147 {
148 if( p.compare( 0, prefix.size(), prefix ) != 0 )
149 {
150 continue;
151 }
152 }
153 }
154
155 if( hassub )
156 {
157 if( p.size() < 2 )
158 {
159 continue;
160 }
161
162 size_t sspos;
163
164 try
165 {
166 sspos = p.find( substr, 1 ); // only match if not prefix
167 }
168 catch( ... )
169 {
170 std::throw_with_nested( err::exceptthrown( "fileUtils::getFileNames",
171 __FILE__,
172 __LINE__,
173 "from std::string::find" ) );
174 }
175
176 if( sspos == std::string::npos )
177 {
178 continue;
179 }
180 }
181
182 // If here then it passed all checks
183 vect.push_back( it->path().native() );
184 }
185
186 sort( vect.begin(), vect.end() );
187 }
188 catch( const err::exceptthrown & e) //this is thrown from within the loop so just pass it on
189 {
190 throw; //don't add anything
191 }
192 catch( const std::filesystem::filesystem_error & e)
193 {
194 std::throw_with_nested( err::exceptthrown( "fileUtils::getFileNames",
195 __FILE__,
196 __LINE__,
197 std::string("from std::filesystem: ") + e.what() ) );
198 }
199 catch( const std::exception & e )
200 {
201 std::throw_with_nested( err::exceptthrown( "fileUtils::getFileNames",
202 __FILE__,
203 __LINE__,
204 std::string("(possibly from std::string) ") + e.what() ) );
205 }
206 catch( ... )
207 {
208 std::throw_with_nested( err::exceptthrown( "fileUtils::getFileNames",
209 __FILE__,
210 __LINE__,
211 "unknown exception" ) );
212 }
213 }
214 else
215 {
216 mxThrowException( err::invalidarg, "fileUtils::getFileNames", directory + " is not a directory" );
217 }
218 }
219 else
220 {
221 mxThrowException( err::notfound, "fileUtils::getFileNames", directory + " does not exist" );
222 }
223
224 return vect;
225}

237std::string fileNamePrependAppend( const std::string &fname, const std::string &prepend, const std::string &append )
238{
239 std::string dir, base, ext;
240
241 std::filesystem::path p = fname;
242 dir = p.parent_path().string();
243 base = p.stem().string();
244 ext = p.extension().string();
245
246 return dir + '/' + prepend + base + append + ext;
247}

259std::string
260getSequentialFilename( const std::string &basename, const std::string &extension, const int startat, const int ndigit )
261{
262 // int maxdig = 1;
263 // for(int j=0;j<ndigit;++j) maxdig *= 10;
264 int maxdig = pow( 10, ndigit );
265
266 char formstr[64];
267 snprintf( formstr, sizeof( formstr ), "%%0%dd", ndigit );
268
269 char digstr[64];
270 int i = startat;
271
272 std::stringstream outn;
273
274 snprintf( digstr, sizeof( digstr ), formstr, i );
275
276 outn << basename;
277 outn << digstr;
278 outn << extension;
279
280 while( std::filesystem::exists( outn.str() ) && i < maxdig )
281 {
282 ++i;
283 outn.str( "" );
284
285 snprintf( digstr, sizeof( digstr ), formstr, i );
286
287 outn << basename;
288 outn << digstr;
289
290 outn << extension;
291 }
292
293 return outn.str();
294}

296off_t fileSize( int fd )
297{
298 if( fd == -1 )
299 {
300 return -1;
301 }
302
303 struct stat stbuf;
304
305 if( ( fstat( fd, &stbuf ) != 0 ) || ( !S_ISREG( stbuf.st_mode ) ) )
306 {
307 return -1;
308 }
309
310 return stbuf.st_size;
311}
