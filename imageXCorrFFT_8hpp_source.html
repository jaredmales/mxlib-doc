45 /// Find the optimum shift to align two images using the FFT cross correlation.
46 /** The reference image must be the same size as the target image. Both the reference image and the section of
47  * target image being analyzed are mean subtracted and variance normalized. An optional mask can be supplied,
48  * which limits the pixels used for the mean and variance calculation.
49  *
50  * Typical usage will be to set the mask, then the reference, then repeatedly call operator() to
51  * determine the shifts for a sequence of imaages. No new heap allocations take place on these calls
52  * to operator(), and the reference image is not re-normalized on each call.
53  *
54  * The shift is reported in pixels such that if the mxlib imageShift function is used
55  * to shift the input image by the negative of the shifts, it will align with the
56  * reference at the center of the array.
57  *
58  * Three peak finding methods are provided. xcorrPeakMethod::centerOfLight uses center of light,
59  * xcorrPeakMethod::gaussFit uses Gaussian centroiding, and xcorrPeakMethod::interpPeak uses interpolation
60  * to find the peak to a given tolerance. For well behaved images gaussFit is most accurate. interpPeak
61  * should be robust against weird shapes and other problems, but accuracy is somewhat worse.
62  *
63  * \tparam _ccImT is the Eigen-like array type used for image processing. See typedefs.
64  *
65  * \ingroup image_reg
66  */

386 template< class ccImT>
387 int imageXCorrFFT<ccImT>::refIm( const ccImT & im )
388 {
389  ccImT im0;
390 
391  //Mask if needed
392  if(m_haveMask)
393  {
394  if(im.rows()!=m_maskIm.rows() && im.cols() != m_maskIm.cols())
395  {
396  mxError("imageXCorFit::setReference", MXE_SIZEERR, "reference and mask are not the same size");
397  return -1;
398  }
399  im0 = im*m_maskIm;
400  }
401  else
402  {
403  im0 = im;
404  }
405 
406  //Setup the FFTW space
407  resize(im0.rows(), im0.cols());
408 
409  //Now normalize
410  if(m_normalize)
411  {
412  realT m = imageMean(im0);
413  realT v = imageVariance(im0, m);
414  m_refIm = (im0 - m)/sqrt(v);
415  }
416  else
417  {
418  m_refIm = im0;
419  }
420  //We save refIm as the un-shifted version
421 
422  //Now shift so center pixel is 0,0
423  imageShiftWP( im0, m_refIm, 0.5*m_rows + 1, 0.5*m_cols + 1 );
424 
425  //Then FT
426  m_fft_fwd(m_ftIm0.data(), im0.data());
427 
428  //Conjugate and normalize for FFTW scaling.
429  for(int c =0; c < m_ftIm0.cols(); ++c)
430  {
431  for(int r=0; r < m_ftIm0.rows(); ++r)
432  {
433  complexT val = std::conj(m_ftIm0(r,c));
434  val /= (m_rows*m_cols);
435  m_ftIm0(r,c) = val;
436  }
437  }
438 
439  //And finally find the reference shift
440  m_refX0 = 0;
441  m_refY0 = 0;
442 
443  operator()(m_refX0, m_refY0, m_refIm);
444 
445  return 0;
446 }

472 template< class ccImT>
473 void imageXCorrFFT<ccImT>::findPeak( Scalar & xShift,
474  Scalar & yShift
475  )
476 {
477  int xLag0,yLag0;
478  Scalar pk = m_ccIm.maxCoeff(&xLag0, &yLag0);
479  Scalar mn = m_ccIm.minCoeff();
480 
481  if(xLag0 - m_maxLag < 0) m_maxLag = xLag0;
482  if(xLag0 + 2*m_maxLag + 1 >= m_ccIm.rows()) m_maxLag = (m_ccIm.rows() - 1 - xLag0)/2;
483  if(yLag0 - m_maxLag < 0) m_maxLag = yLag0;
484  if(yLag0 + 2*m_maxLag + 1 >= m_ccIm.cols()) m_maxLag = (m_ccIm.cols() - 1 - yLag0)/2;
485 
486  realT x0 = xLag0-m_maxLag;
487  realT y0 = yLag0-m_maxLag;
488 
489  if(m_peakMethod == xcorrPeakMethod::gaussFit)
490  {
491  m_magIm = m_ccIm.block( x0, y0, 2*m_maxLag+1, 2*m_maxLag+1);
492 
493  m_fitter.setArray(m_magIm.data(), m_magIm.rows(), m_magIm.cols());
494  m_magIm.maxCoeff(&xLag0, &yLag0);
495 
496  m_fitter.setGuess(mn, pk-mn, m_maxLag, m_maxLag, 3, 3, 0);
497  m_fitter.fit();
498 
499  xShift = (m_fitter.x0() + x0) - m_refX0;
500  yShift = (m_fitter.y0() + y0) - m_refY0;;
501  }
502  else if(m_peakMethod == xcorrPeakMethod::interpPeak)
503  {
504  int x, y;
505  m_magIm.resize(m_magSize, m_magSize);
506 
507  imageMagnify(m_magIm, m_ccIm.block(x0, y0, 2*m_maxLag+1, 2*m_maxLag+1), cubicConvolTransform<Scalar>());
508 
509  m_magIm.maxCoeff(&x,&y);
510  xShift = (x*m_tol + x0) - m_refX0;
511  yShift = (y*m_tol + y0) - m_refY0;
512  }
513  else if(m_peakMethod == xcorrPeakMethod::centerOfLight)
514  {
515  Scalar x,y;
516 
517  m_magIm = m_ccIm.block( x0, y0, 2*m_maxLag+1, 2*m_maxLag+1);
518  m_magIm -= m_magIm.minCoeff(); //Must sum to > 0.
519 
520  imageCenterOfLight(x,y,m_magIm);
521 
522  xShift = (x + x0) - m_refX0;
523  yShift = (y + y0) - m_refY0;
524  }
525  else if(m_peakMethod == xcorrPeakMethod::none)
526  {
527  int x,y;
528  m_ccIm.maxCoeff(&x, &y);
529  xShift = x;
530  yShift = y;
531  }
532  else
533  {
534  mxThrowException(mx::err::invalidconfig, "imageXCorrFFT<ccImT>::operator()", "unknown peak finding method");
535  }
536 
537 }
