39  explicit gaussKernel(arithT fwhm)
40  {
41  _fwhm = fwhm;
42 
43  int w = kernW*_fwhm;
44 
45  if(w % 2 == 0) w++;
46 
47  kernel.resize(w, w);
48 
49  arithT kcen = 0.5*(w-1.0);
50 
51  arithT sig2 = _fwhm/2.354820045030949327;
52  sig2 *= sig2;

88 template<typename _arrayT, size_t _kernW=4>
89 struct azBoxKernel

98  arithT _azWidth;

101  azBoxKernel( arithT radWidth, ///< [in] the half-width of the averaging box, in the radial direction, in pixels.
102  arithT azWidth ///< [in] the half-width of the averaging box, in the azimuthal direction, in pixels.

116  void setKernel(arithT x, arithT y, arrayT & kernel)
117  {
118  arithT rad = sqrt((arithT) (x*x + y*y));
119 
120  arithT sinq = y/rad;
121  arithT cosq = x/rad;
122 
123 
124  int w = 2*( (int) std::max( fabs(_azWidth*sinq), fabs(_radWidth*cosq) ) + 1 );
125  int h = 2*( (int) std::max( fabs(_azWidth*cosq), fabs(_radWidth*sinq) ) + 1);
126 
127 
128  kernel.resize(w, h);
129 
130  arithT xcen = 0.5*(w-1.0);
131  arithT ycen = 0.5*(h-1.0);
132 
133  arithT sinq2, cosq2, sindq;
134  for(int i=0; i < w; ++i)
135  {
136  arithT xP = i -0.5;
137  for(int j=0; j < h; ++j)
138  {
139  arithT yP = j-0.5;
140  arithT radP = sqrt( pow(x+xP-xcen,2) + pow(y+yP-ycen,2) );
141  if( fabs(rad-radP) > _radWidth)
142  {
143  kernel(i,j) = 0;
144  continue;
145  }
146 
147  sinq2 = (yP-ycen)/radP;
148  cosq2 = (xP-xcen)/radP;
149  sindq = sinq2*cosq - cosq2*sinq;
150  if( fabs(radP*sindq) <= _azWidth) kernel(i,j) = 1;
151  else kernel(i,j) = 0;
152  }
153  }
154  if(kernel.sum() == 0)
155  {
156  std::cerr << "Kernel sum 0: " << x << " " << y << "\n";
157  //exit(-1);
158  }
159  kernel /= kernel.sum();
160  }

830 /// Form a standard deviation image, and optionally divide the input by it to form a S/N map.
831 /** The standard deviation profile is calculated using linear interpolation on a 1 pixel grid
832  *
833  * \tparam eigenImT the eigen array type of the output and non-reference images. Each image input can be a different type to allow references, etc.
834  *
835  */
836 template<typename eigenImT, typename eigenImT1, typename eigenImT2, typename eigenImT3>
837 void stddevImage( eigenImT & stdIm, ///< [out] the standard deviation image. This will be resized.
838  const eigenImT1 & im, ///< [in] the image to form the standard deviation profile of, never altered.
839  const eigenImT2 & rad, ///< [in] array of radius values
840  const eigenImT3 & mask, ///< [in] a 1/0 mask. 0 pixels are excluded from the std-dev calculations.
841  typename eigenImT::Scalar minRad, ///< [in] the minimum radius to analyze
842  typename eigenImT::Scalar maxRad, ///< [in] the maximum radius to analyze
843  bool divide ///< [in] if true, the output is the input image is divided by the std-dev profile, i.e. a S/N map. default is false.
844  )
845 {
846  typedef typename eigenImT::Scalar floatT;
847 
848  int dim1 = im.cols();
849  int dim2 = im.rows();
850 
851  floatT mr = rad.maxCoeff();
852 
853  /* A vector of radvals will be sorted, then binned*/
854  std::vector<radval<floatT> > rv(dim1*dim2);
855 
856  for(int i=0;i<rv.size();++i)
857  {
858  if(mask(i) == 0) continue;
859 
860  rv[i].r = rad(i);
861  rv[i].v = im(i);
862  }
863 
864  sort(rv.begin(), rv.end(), radvalRadComp<floatT>());
865 
866  /*Now bin*/
867  floatT dr = 1;
868  floatT r0 = 0;
869  floatT r1 = dr;
870  int i1=0, i2, n;
871 
872  floatT stdVal;
873 
874  std::vector<double> std_r, std_v;
875 
876  while(r1 < mr)
877  {
878  while(rv[i1].r < r0 && i1 < rv.size())
879  {
880  ++i1;
881  }
882  if(i1 == rv.size())
883  {
884  break;
885  }
886 
887  i2 = i1;
888  while(rv[i2].r <= r1 && i2 < rv.size()) ++i2;
889  n = 0.5*(i2-i1);
890 
891  std::vector<double> vals;
892 
893  for(int i=i1; i< i2; ++i)
894  {
895  vals.push_back( rv[i].v);
896  }
897 
898  std_r.push_back(.5*(r0+r1));
899 
900  std_v.push_back( std::sqrt(math::vectorVariance(vals)) ) ;
901  i1 = i2;
902  r0 += dr;
903  r1 += dr;
904  }
905 
906  /* And finally, interpolate onto the radius image */
907  stdIm.resize(dim1, dim2);
908  mx::gslInterpolator<double> interp(gsl_interp_linear, std_r, std_v);
909 
910  for(int i=0;i<dim1;++i)
911  {
912  for(int j=0;j<dim2;++j)
913  {
914  if(rad(i,j) < minRad || rad(i,j) > maxRad)
915  {
916  stdIm(i,j) = 0;
917  }
918  else
919  {
920  stdIm(i,j) = interp.interpolate( ((double) rad(i,j)) );
921  if(divide) stdIm(i,j) = im(i,j) / stdIm(i,j);
922  }
923  }
924  }
925 
926 }
