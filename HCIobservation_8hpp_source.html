1204template <typename _realT>
1205void HCIobservation<_realT>::coaddImages( int coaddCombineMethod,
1206 int coaddMaxImno,
1207 int coaddMaxTime,
1208 std::vector<std::string> &coaddKeywords,
1209 std::vector<double> &imageMJD,
1210 std::vector<fits::fitsHeader> &heads,
1211 eigenCube<realT> &ims )
1212{
1213 std::cerr << "***************************************************************\n";
1214 std::cerr << " *** WARNING *** \n";
1215 std::cerr << " coadding is poorly tested. proceed with caution. \n";
1216 std::cerr << "***************************************************************\n";
1217
1218 // Validate setup
1219 if( coaddMaxImno <= 0 && coaddMaxTime <= 0 )
1220 return;
1221
1222 // Validate combine method
1223 if( coaddCombineMethod == HCI::noCombine )
1224 return;
1225
1226 int Nims = ims.planes();
1227 int Nrows = ims.rows();
1228 int Ncols = ims.cols();
1229
1230 t_coadd_begin = sys::get_curr_time();
1231
1232 std::vector<eigenImageT> coadds;
1233
1234 // We do all math here in double, to avoid losing precision
1235 std::vector<double> avgMJD;
1236 std::vector<std::vector<double>> avgVals;
1237
1238 int combineMethod = HCI::medianCombine;
1239 if( coaddCombineMethod == HCI::meanCombine )
1240 combineMethod = HCI::meanCombine;
1241
1242 // Index range of images for next coadd
1243 int im0, imF;
1244 im0 = 0;
1245 imF = 1;
1246
1247 // Accumulate images to coadd into a cube
1248 eigenCube<realT> imsToCoadd;
1249
1250 // Temporary for combination.
1251 eigenImageT coadd;
1252
1253 // Accumulate values
1254 double initMJD;
1255 std::vector<double> initVals;
1256 initVals.resize( coaddKeywords.size() );
1257
1258 while( im0 < Nims )
1259 {
1260 // Initialize the accumulators
1261 initMJD = imageMJD[im0];
1262
1263 for( size_t i = 0; i < coaddKeywords.size(); ++i )
1264 {
1265 initVals[i] = heads[im0][coaddKeywords[i]].value<double>();
1266 }
1267
1268 // Now increment imF, then test whether each variable is now outside the range
1269 bool increment = true;
1270 while( increment == true )
1271 {
1272 ++imF;
1273
1274 if( imF >= Nims )
1275 {
1276 imF = Nims;
1277 increment = false;
1278 break;
1279 }
1280
1281 if( imF - im0 > coaddMaxImno && coaddMaxImno > 0 )
1282 {
1283 --imF;
1284 increment = false;
1285 break;
1286 }
1287
1288 ///\todo this should really include end of exposure too.
1289 if( ( imageMJD[imF] - imageMJD[im0] ) * 86400.0 > coaddMaxTime && coaddMaxTime > 0 )
1290 {
1291 --imF;
1292 increment = false;
1293 break;
1294 }
1295
1296 } // while(increment == true)
1297 // At this point, imF is the first image NOT to include in the next coadd.
1298
1299 // Now actually do the accumulation
1300 ///\todo this needs to handle averaging of angles
1301 for( int imno = im0 + 1; imno < imF; ++imno )
1302 {
1303 initMJD += imageMJD[imno];
1304
1305 for( size_t i = 0; i < coaddKeywords.size(); ++i )
1306 {
1307 initVals[i] += heads[imno][coaddKeywords[i]].value<double>();
1308 }
1309 }
1310
1311 // And then turn them into an average
1312 initMJD /= ( imF - im0 );
1313 for( size_t i = 0; i < coaddKeywords.size(); ++i )
1314 {
1315 initVals[i] /= ( imF - im0 );
1316 }
1317
1318 // Extract the images into the temporary
1319 imsToCoadd.resize( Nrows, Ncols, imF - im0 );
1320 for( int i = 0; i < ( imF - im0 ); ++i )
1321 {
1322 imsToCoadd.image( i ) = ims.image( im0 + i );
1323 }
1324
1325 // Here do the combine and insert into the vector
1326 if( combineMethod == HCI::medianCombine )
1327 {
1328 imsToCoadd.median( coadd );
1329 }
1330 if( combineMethod == HCI::meanCombine )
1331 {
1332 imsToCoadd.mean( coadd );
1333 }
1334 coadds.push_back( coadd );
1335
1336 // Insert the new averages
1337 avgMJD.push_back( initMJD );
1338 avgVals.push_back( initVals );
1339
1340 im0 = imF;
1341 imF = im0 + 1;
1342 } // while(im0 < Nims)
1343
1344 // Now resize ims and copy the coadds to the new cube
1345 ims.resize( Nrows, Ncols, coadds.size() );
1346 Nims = coadds.size();
1347
1348 for( int i = 0; i < Nims; ++i )
1349 ims.image( i ) = coadds[i];
1350
1351 // Now deal with imageMJD and headers
1352 imageMJD.erase( imageMJD.begin() + Nims, imageMJD.end() );
1353 heads.erase( heads.begin() + Nims, heads.end() );
1354
1355 for( int i = 0; i < Nims; ++i )
1356 {
1357 imageMJD[i] = avgMJD[i];
1358 for( size_t j = 0; j < coaddKeywords.size(); ++j )
1359 {
1360 heads[i][coaddKeywords[j]].value( avgVals[i][j] );
1361 }
1362 }
1363
1364 t_coadd_end = sys::get_curr_time();
1365

1476 medmask.col( medmask.cols() - 1 - z ) = 0;
1477 }
1478
1479#pragma omp parallel
1480 {
1481 eigenImageT fim, im;
1482 fim.resize( ims.rows(), ims.cols() );

1485#pragma omp for
1486 for( int i = 0; i < ims.planes(); ++i )
1487 {
1488 im = ims.image( i );
1489 medianSmooth( fim, im, m_preProcess_medianUSM_fwhm );

1491 }
1492 }
1493
1494 if( m_maskFile != "" && m_preProcess_mask )
1495 {
1496 std::cerr << "Masking . . .\n";
1497#pragma omp parallel for
1498 for( int i = 0; i < ims.planes(); ++i )
1499 {
1500 ims.image( i ) *= m_mask;
1501 }
1502 }
1503 std::cerr << "done\n";
1504 }

1507 {
1508 std::cerr << "applying Gauss USM . . .\n";
1509 t_gaussusm_begin = sys::get_curr_time();
1510
1511#pragma omp parallel for
1512 for( int i = 0; i < ims.planes(); ++i )
1513 {
1514 eigenImageT fim, im;
1515 im = ims.image( i );
1516 filterImage( fim,
1517 im,
1518 gaussKernel<eigenImage<_realT>, 2>( m_preProcess_gaussUSM_fwhm ),
1519 0.5 * ( ims.cols() - 1 ) - m_preProcess_gaussUSM_fwhm * 4 );
1520 im = ( im - fim );
1521 ims.image( i ) = im;
1522 }
1523
1524 if( m_maskFile != "" && m_preProcess_mask )
1525 {
1526 std::cerr << "Masking . . .\n";
1527#pragma omp parallel for
1528 for( int i = 0; i < ims.planes(); ++i )
1529 {
1530 ims.image( i ) *= m_mask;

1532 }
1533 t_gaussusm_end = sys::get_curr_time();

1537 if( m_preProcess_azUSM_azW && m_preProcess_azUSM_radW )

1539 ipc::ompLoopWatcher<> status( ims.planes(), std::cerr );

1541 std::cerr << "applying azimuthal USM . . .\n";
1542 t_azusm_begin = sys::get_curr_time();
1543#pragma omp parallel for
1544 for( int i = 0; i < ims.planes(); ++i )
1545 {
1546 eigenImageT fim, im;
1547 im = ims.image( i );
1548 medianFilterImage( fim,
1549 im,
1550 azBoxKernel<eigenImage<realT>>(
1551 m_preProcess_azUSM_radW, m_preProcess_azUSM_azW, m_preProcess_azUSM_maxAz ) );
1552 im = ( im - fim );
1553 ims.image( i ) = im;
1554 status.incrementAndOutputStatus();
1555 }
1556
1557 if( m_maskFile != "" && m_preProcess_mask )
1558 {
1559 std::cerr << "Masking . . .\n";
1560#pragma omp parallel for
1561 for( int i = 0; i < ims.planes(); ++i )
1562 {
1563 ims.image( i ) *= m_mask;
1564 }
1565 }
1566
1567 t_azusm_end = sys::get_curr_time();
1568 std::cerr << "done (" << t_azusm_end - t_azusm_begin << " sec) \n";
1569 }
1570
1571 t_preproc_end = sys::get_curr_time();
