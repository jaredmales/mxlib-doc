363 template<typename _realT, class _derotFunctObj, typename _evCalcT>
364 void KLIPreduction<_realT, _derotFunctObj, _evCalcT>::meanSubtract( eigenCube<realT> & ims,
365  eigenCube<realT> & tims,
366  std::vector<_realT> & norms
367  )
368 {
369 
370  norms.resize(ims.planes());
371 
372  if(m_meanSubMethod == HCI::meanImage || m_meanSubMethod == HCI::medianImage)
373  {
374  eigenImageT mean;
375 
376  if(m_meanSubMethod == HCI::meanImage)
377  {
378  ims.mean(mean);
379  }
380  else if(m_meanSubMethod == HCI::medianImage)
381  {
382  ims.median(mean);
383  }
384 
385  for(int n=0;n<ims.planes(); ++n)
386  {
387  ims.image(n) -= mean;
388 
389  realT immean = ims.image(n).mean();
390  norms[n] = (ims.image(n)-immean).matrix().norm();
391  }
392 
393  if(&tims != &ims)
394  {
395  for(int n=0;n<tims.planes(); ++n)
396  {
397  tims.image(n) -= mean;
398  }
399  }
400  }
401  else
402  {
403  realT mean;
404  std::vector<realT> work; //Working memmory for median calc
405 
406  for(int n=0;n<ims.planes(); ++n)
407  {
408  if( m_meanSubMethod == HCI::imageMean )
409  {
410  mean = ims.image(n).mean();
411  }
412  else if(m_meanSubMethod == HCI::imageMedian)
413  {
414  mean = imageMedian(ims.image(n), &work);
415  }
416 
417  ims.image(n) -= mean;
418 
419  //Because we might not have used the mean, we need to re-mean to make this the standard deviation
420  realT immean = ims.image(n).mean();
421  norms[n] = (ims.image(n)-immean).matrix().norm();
422 
423  }
424 
425  if(&tims != &ims)
426  {
427  for(int n=0;n<tims.planes(); ++n)
428  {
429  if( m_meanSubMethod == HCI::imageMean )
430  {
431  mean = tims.image(n).mean();
432  }
433  else if(m_meanSubMethod == HCI::imageMedian)
434  {
435  mean = imageMedian(tims.image(n), &work);
436  }
437 
438  tims.image(n) -= mean;
439  }
440  }
441  }
442 }

652 template<typename realT, typename eigenT, typename eigenTv, class derotFunctObj>
653 void collapseCovar( eigenT & cutCV,
654  const eigenT & CV,
655  const std::vector<realT> & sds,
656  eigenT & rimsCut,
657  const eigenTv & rims,
658  int imno,
659  double dang,
660  double dangMax,
661  int Nims,
662  int excludeMethod,
663  int excludeMethodMax,
664  int includeRefNum,
665  const derotFunctObj & derotF,
666  eigenImage<int> & imsIncluded
667  )
668 {
669  std::vector<cvEntry> allidx(Nims);
670 
671  std::cerr << "dangs: " << dang << " " << dangMax << "\n";
672 
673 
674  //Initialize the vector of cvEntries
675  for(int i=0; i < Nims; ++i)
676  {
677  allidx[i].index = i;
678  allidx[i].angle = derotF.derotAngle(i);
679 
680  //CV is lower-triangular
681  if(i <= imno)
682  {
683  allidx[i].cvVal = CV(imno,i)/(sds[i]*sds[imno]);
684  }
685  else
686  {
687  allidx[i].cvVal = CV(i,imno)/(sds[i]*sds[imno]);
688  }
689  }
690 
691  if(excludeMethod == HCI::excludePixel || excludeMethod == HCI::excludeAngle )
692  {
693  for(size_t j=0; j < Nims; ++j)
694  {
695  if( fabs(math::angleDiff<math::radians>( derotF.derotAngle(j), derotF.derotAngle(imno))) <= dang ) allidx[j].included = false;
696  }
697  }
698  else if(excludeMethod == HCI::excludeImno)
699  {
700  for(size_t j=0; j < Nims; ++j)
701  {
702  if( fabs( (long) j - imno) <= dang ) allidx[j].included = false;
703  }
704  }
705 
706  if(excludeMethodMax == HCI::excludePixel || excludeMethodMax == HCI::excludeAngle )
707  {
708  for(size_t j=0; j < Nims; ++j)
709  {
710  if( fabs(math::angleDiff<math::radians>( derotF.derotAngle(j), derotF.derotAngle(imno))) > dangMax ) allidx[j].included = false;
711  }
712  }
713  else if(excludeMethodMax == HCI::excludeImno)
714  {
715  for(size_t j=0; j < Nims; ++j)
716  {
717  if( fabs( (long) j - imno) > dangMax ) allidx[j].included = false;
718  }
719  }
720 
721 
722  if( includeRefNum > 0 && (size_t) includeRefNum < allidx.size())
723  {
724  long kept = 0;
725  for(size_t j=0; j < Nims; ++j)
726  {
727  if(allidx[j].included == true) ++kept;
728  }
729 
730  //Get a vector for sorting
731  std::vector<realT> cvVal;
732  cvVal.resize(kept);
733  size_t k = 0;
734  for(size_t j=0; j < Nims; ++j)
735  {
736  if(allidx[j].included == true)
737  {
738  cvVal[k] = allidx[j].cvVal;
739  ++k;
740  }
741  }
742 
743  //Partially sort the correlation values
744  std::nth_element(cvVal.begin(), cvVal.begin()+ (kept-includeRefNum), cvVal.end());
745 
746  realT mincorr = cvVal[kept-includeRefNum];
747  std::cerr << " Minimum correlation: " << mincorr << "\n";
748 
749 
750  for(size_t j=0; j < Nims; ++j)
751  {
752  if( allidx[j].cvVal < mincorr ) allidx[j].included = false;
753  }
754 
755  }
756 
757  std::vector<size_t> keepidx;
758  for(size_t j=0;j<Nims;++j)
759  {
760  imsIncluded(imno,j) = allidx[j].included;
761 
762  if(allidx[j].included) keepidx.push_back(j);
763  }
764 
765  std::cerr << " Keeping " << keepidx.size() << " reference images out of " << Nims << " (" << Nims-keepidx.size() << " rejected)\n";
766 
767  if(keepidx.size() == 0)
768  {
769  std::cerr << "\n\n" << imno << "\n\n";
770  }
771 
772  extractRowsAndCols(cutCV, CV, keepidx);
773  extractCols(rimsCut, rims, keepidx);
774 
775 }
