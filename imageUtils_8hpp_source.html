126  )

145  typename imageT::Scalar & y,

169 /// Find the maximum in an image at sub-pixel resolution by interpolation
170 /** Uses imageMagnify() to expand the image to the desired scale. Because of the
171  * scaling used in imageMagnify, the desired scale may not be exact. As a result
172  * the actual scale is returned in scale_x and scale_y.
173  *
174  */
175 template<typename floatT, typename imageT, typename magImageT, typename transformT>
176 int imageMaxInterp( floatT & x, ///< [out] the x-position of the maximum, in pixels of the input image
177  floatT & y, ///< [out] the y-position of the maximum, in pixels of the input image
178  floatT & scale_x, ///< [in/out] the desired scale or resolution, in pixels < 1, in the x direction. On output contains the actual scale calculated.
179  floatT & scale_y, ///< [in/out] the desired scale or resolution, in pixels < 1, in the y direction. On output contains the actual scale calculated.
180  magImageT & magIm, ///< [in] the magnified image. This is used as working memory, will be resized.
181  const imageT & im, ///< [in] the image to find the maximum of
182  transformT trans ///< [in] the transform to use for interpolation
183  )
184 {
185  floatT magSize_x = ceil( (im.rows() - 1.0)/ scale_x) + 1;
186  floatT magSize_y = ceil( (im.cols() - 1.0)/ scale_y) + 1;
187 
188  floatT mag_x = ((floatT) magSize_x-1.0)/((floatT) im.rows() - 1.0);
189  floatT mag_y = ((floatT) magSize_y-1.0)/((floatT) im.cols() - 1.0);
190 
191  scale_x = 1.0/mag_x;
192  scale_y = 1.0/mag_y;
193 
194  magIm.resize(magSize_x, magSize_y);
195 
196  imageMagnify(magIm, im, trans);
197 
198  int ix, iy;
199  magIm.maxCoeff(&ix,&iy);
200  x = ix*scale_x;
201  y = iy*scale_y;
202 
203  return 0;
204 }

227 /// Combine two images, each with their own mask defining good pixels.
228 /** The combined image is made up of the pixels in im1 where mask1 is 1, and the pixels of im2 where mask2 is 1.
229  * If a pixel in both mask1 and mask2 has a value of 1, that pixel in the combo is the average of im1 and im2.
230  * All other pixels are set to 0 in the combined image.
231  *
232  * Separate template types are used for each argument to allow references, etc.
233  *
234  * \tparam imageT the eigen-like array type of the combined image
235  * \tparam imageT1 the eigen-like array type of image 1
236  * \tparam imageT2 the eigen-like array type of mask 1
237  * \tparam imageT3 the eigen-like array type of image 2
238  * \tparam imageT4 the eigen-like array type of mask 2
239  */
240 template<typename imageT, typename imageT1, typename imageT2, typename imageT3, typename imageT4>
241 void combine2ImagesMasked( imageT & combo, ///< [out] the combined image. will be resized.
242  const imageT1 & im1, ///< [in] the first image
243  const imageT2 & mask1, ///< [in] the mask for the first image
244  const imageT3 & im2, ///< [in] the second image
245  const imageT4 & mask2 ///< [in] the mask for the second image
246  )
247 {
248  combo.resize(im1.rows(), im2.cols());
249 
250  for(int c=0; c < combo.cols(); ++c)
251  {
252  for(int r=0;r<combo.rows();++r)
253  {
254  if(mask1(r,c) == 1 && mask2(r,c) == 0) combo(r,c) = im1(r,c);
255  else if(mask2(r,c) == 1 & mask1(r,c) == 0) combo(r,c) = im2(r,c);
256  else if(mask1(r,c) == 1 && mask2(r,c) == 1) combo(r,c) = 0.5*(im1(r,c) + im2(r,c));
257  else combo(r,c) = 0;
258  }
259  }
260 }
