100 /// Calculate the variance of an image w.r.t. a given value
101 /**
102  *
103  * \returns the variance of the input image
104  */
105 template< class imageT>
106 typename imageT::Scalar imageVariance( imageT & im, ///< [in] the image of which to calculate the variance
107  typename imageT::Scalar mean ///< [in] the value to calculate the variance with respect to
108  )
109 {
110  typename imageT::Scalar v = 0;
111 
112  for(int c=0;c<im.cols();++c)
113  {
114  for(int r=0;r<im.rows();++r)
115  {
116  v += pow(im(r,c)-mean,2);
117  }
118  }
119 
120  v /= (im.rows()*im.cols());
121 
122  return v;

151 /// Fine the maximum in an image at sub-pixel resolution by interpolation
152 /** Uses imageMagnify() to expand the image to the desired scale. Because of the
153  * scaling used in imageMagnify, the desired scale may not be exact. As a result
154  * the actual scale is returned in scale_x and scale_y.
155  *
156  */
157 template<typename floatT, typename imageT, typename magImageT, typename transformT>
158 int imageMaxInterp( floatT & x, ///< [out] the x-position of the maximum, in pixels of the input image
159  floatT & y, ///< [out] the y-position of the maximum, in pixels of the input image
160  floatT & scale_x, ///< [in/out] the desired scale or resolution, in pixels < 1, in the x direction. On output contains the actual scale calculated.
161  floatT & scale_y, ///< [in/out] the desired scale or resolution, in pixels < 1, in the y direction. On output contains the actual scale calculated.
162  magImageT & magIm, ///< [in] the magnified image. This is used as working memory, will be resized.
163  const imageT & im, ///< [in] the image to find the maximum of
164  transformT trans ///< [in] the transform to use for interpolation
165  )
166 {
167  floatT magSize_x = ceil( (im.rows() - 1.0)/ scale_x) + 1;
168  floatT magSize_y = ceil( (im.cols() - 1.0)/ scale_y) + 1;
169 
170  floatT mag_x = ((floatT) magSize_x-1.0)/((floatT) im.rows() - 1.0);
171  floatT mag_y = ((floatT) magSize_y-1.0)/((floatT) im.cols() - 1.0);
172 
173  scale_x = 1.0/mag_x;
174  scale_y = 1.0/mag_y;
175 
176  magIm.resize(magSize_x, magSize_y);
177 
178  imageMagnify(magIm, im, trans);
179 
180  int ix, iy;
181  magIm.maxCoeff(&ix,&iy);
182  x = ix*scale_x;
183  y = iy*scale_y;
184 
185  return 0;
186 }

206 /// Combine two images, each with their own mask defining good pixels.
207 /** The combined image is made up of the pixels in im1 where mask1 is 1, and the pixels of im2 where mask2 is 1.
208  * If a pixel in both mask1 and mask2 has a value of 1, that pixel in the combo is the average of im1 and im2.
209  * All other pixels are set to 0 in the combined image.
210  *
211  * Separate template types are used for each argument to allow references, etc.
212  *
213  * \tparam imageT the eigen-like array type of the combined image
214  * \tparam imageT1 the eigen-like array type of image 1
215  * \tparam imageT2 the eigen-like array type of mask 1
216  * \tparam imageT3 the eigen-like array type of image 2
217  * \tparam imageT4 the eigen-like array type of mask 2
218  */
219 template<typename imageT, typename imageT1, typename imageT2, typename imageT3, typename imageT4>
220 void combine2ImagesMasked( imageT & combo, ///< [out] the combined image. will be resized.
221  const imageT1 & im1, ///< [in] the first image
222  const imageT2 & mask1, ///< [in] the mask for the first image
223  const imageT3 & im2, ///< [in] the second image
224  const imageT4 & mask2 ///< [in] the mask for the second image
225  )
226 {
227  combo.resize(im1.rows(), im2.cols());
228 
229  for(int c=0; c < combo.cols(); ++c)
230  {
231  for(int r=0;r<combo.rows();++r)
232  {
233  if(mask1(r,c) == 1 && mask2(r,c) == 0) combo(r,c) = im1(r,c);
234  else if(mask2(r,c) == 1 & mask1(r,c) == 0) combo(r,c) = im2(r,c);
235  else if(mask1(r,c) == 1 && mask2(r,c) == 1) combo(r,c) = 0.5*(im1(r,c) + im2(r,c));
236  else combo(r,c) = 0;
237  }
238  }
239 }
