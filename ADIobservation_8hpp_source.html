513 template<typename _realT, class _derotFunctObj>
514 int ADIobservation<_realT, _derotFunctObj>::injectFake( eigenImageT & fakePSF,
515  eigenCube<realT> & ims,
516  int image_i,
517  realT derotAngle,
518  realT PA,
519  realT sep,
520  realT contrast,
521  realT scale,
522  realT RDIFluxScale,
523  realT RDISepScale
524  )
525 {
526 
527  //Check for correct sizing
528  if( (fakePSF.rows() < ims.rows() && fakePSF.cols() >= ims.cols()) ||
529  (fakePSF.rows() >= ims.rows() && fakePSF.cols() < ims.cols()))
530  {
531  throw mxException("mxlib:high contrast imaging", -1, "image wrong size", __FILE__, __LINE__, "fake PSF has different dimensions and can't be sized properly");
532  }
533 
534  //Check if fake needs to be padded out
535  if(fakePSF.rows() < ims.rows() && fakePSF.cols() < ims.cols())
536  {
537  eigenImageT pfake(ims.rows(), ims.cols());
538  padImage(pfake, fakePSF, 0.5*(ims.rows()-fakePSF.rows()), 0);
539  fakePSF = pfake;
540  }
541 
542  //Check if fake needs to be cut down
543  if(fakePSF.rows() > ims.rows() && fakePSF.cols() > ims.cols())
544  {
545  eigenImageT cfake(ims.rows(), ims.cols());
546  cutPaddedImage(cfake, fakePSF, 0.5*(fakePSF.rows() - ims.rows()));
547  fakePSF = cfake;
548  }
549 
550  /*** Now shift to the separation and PA, scale, apply contrast, and inject ***/
551  //allocate shifted fake psf
552  eigenImageT shiftFake(fakePSF.rows(), fakePSF.cols());
553 
554  realT ang, dx, dy;
555 
556  ang = math::dtor(-1*PA) + derotAngle;
557 
558  dx = sep * RDISepScale * sin(ang);
559  dy = sep * RDISepScale * cos(ang);
560 
561  imageShift(shiftFake, fakePSF, dx, dy, cubicConvolTransform<realT>());
562 
563  ims.image(image_i) = ims.image(image_i) + shiftFake*scale*RDIFluxScale*contrast;
564 
565  return 0;
566 
567 }
