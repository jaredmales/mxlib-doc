48 /// Class to manage a set of configurable values, and read their values from config/ini files and the command line.
49 /**
50  * The configuration files are ini-style, with sections. That is
51  \verbatim
52  key1=value1
53  key2=value2
54 
55  [section1]
56  key3=value3
57  key4=value4,value4.1, value4.2, value4.3
58 
59  [section2]
60  key3=value5
61  key3=value5.1
62  key4=value6_over_
63  multiple_lines
64 
65  \endverbatim
66  * such that section1.key3 is distinct from section2.key3 (they must have different config-target names though).
67  *
68  * Additional syntax rules:
69  * - Leading whitespace is stripped from the value, so `key=val` and `key= val` are equivalent.
70  * - Additional entries within one file with the same section and key are appended to the previous entry.
71  * So the value of section2.key3 is "value5value5.1".
72  * - Multi-line values are handled such that in the above example the result is key4=value6_over_multiple_lines.
73  * - Vectors are input as comma separated lists, as in section1.key4 above. Leading whitespace is stripped from each
74  * component of the vector.
75  *
76  * \todo add handling of += in subsequent files.
77  *
78  * The command line parser handles both short-opt ("-h -vArg -n Arg") and long-opt ("--help --value=Arg --number=Arg") styles.
79  *
80  *
81  * \bug a config=value pair listed in a conf file twice seems to cause a failure, even if they are the same value.
82  *
83  * \ingroup mxApp
84  *
85  */

421 inline
422 void appConfigurator::parseCommandLine( int argc,
423  char ** argv,
424  const std::string & oneTarget
425  )
426 {
427  if(argc == 0) return;
428 
429  clOptions clOpts;
430 
431  //First load the options into the clOptions parser
432  targetIterator it;
433  for(it = m_targets.begin(); it != m_targets.end(); ++it)
434  {
435  if(it->second.shortOpt == "" && it->second.longOpt == "") continue; //No command line opts specified.
436 
437  clOpts.add(it->second.name,it->second.shortOpt.c_str(),it->second.longOpt.c_str(), it->second.clType);
438  }
439 
440  //Then load the command-line-only options.
441  clOnlyTargetIterator cloit;
442  for(cloit = clOnlyTargets.begin(); cloit != clOnlyTargets.end(); ++cloit)
443  {
444  if(cloit->shortOpt == "" && cloit->longOpt == "") continue; //Nothing to add?
445 
446  clOpts.add(cloit->name,cloit->shortOpt.c_str(),cloit->longOpt.c_str(), cloit->clType);
447  }
448 
449  //Now we parse
450  clOpts.parse(argc, argv, &nonOptions);
451 
452  if(clOpts.numUnknown() > 0)
453  {
454  std::vector<std::string> unk;
455  clOpts.unknown(unk);
456 
457  for (size_t n=0;n<unk.size();++n)
458  {
459  //Insert or update existing
460  m_unusedConfigs[unk[n]].name = unk[n];
461  if(m_sources) m_unusedConfigs[unk[n]].sources.push_back("command line");
462  m_unusedConfigs[unk[n]].set = true;
463  }
464  }
465 
466  //If nothing more to do, get out
467  if(clOpts.nOpts == 0)
468  {
469  return;
470  }
471  //And then load the results in the config target map.
472  for(it = m_targets.begin(); it != m_targets.end(); ++it)
473  {
474  if(oneTarget != "" && it->second.name != oneTarget) continue;
475 
476  if(clOpts.optSet(it->second.name))
477  {
478  std::vector<std::string> args;
479 
480  clOpts.getAll(args, it->second.name);
481 
482  it->second.values.insert( it->second.values.end(), args.begin(), args.end());
483  if(m_sources)
484  {
485  for(size_t n=0; n < args.size(); ++n) it->second.sources.push_back("command line");
486  }
487 
488  it->second.verbosity = clOpts.count(it->second.name);
489  it->second.set = true;
490  }
491  }

496 inline
497 int appConfigurator::readConfig( const std::string & fname,
498  bool reportFileNotFound
499  )
500 {
501  //Handle empty string quietly
502  if(fname == "") return 0;
503 
504  iniFile iF;
505 
506  ///\todo update error handling to include >0 (line numer of parse error) and -2 memory allocation error.
507  int prv = iF.parse(fname);
508 
509  if( prv == -1)
510  {
511  if(!reportFileNotFound) return -1;
512 
513  mxError("appConfigurator: ", MXE_FILENOTFOUND, "The file " + fname + " was not found");
514  return -1;
515  }
516 
517  if( prv == -2)
518  {
519  mxError("appConfigurator: ", MXE_ALLOCERR, "Memory allocation error in config file parser");
520  return -1;
521  }
522 
523  if( prv > 0)
524  {
525  mxError("appConfigurator: ", MXE_PARSEERR, "Parsing error in " + fname + " at line " + std::to_string(prv));
526  return -1;
527  }
528 
529  targetIterator it;
530 
531  for(it = m_targets.begin(); it != m_targets.end(); ++it)
532  {
533  if(iF.count(it->second.section, it->second.keyword) > 0)
534  {
535  it->second.values.push_back(iF(it->second.section, it->second.keyword));
536  if(m_sources)
537  {
538  it->second.sources.push_back(fname);
539  }
540  it->second.set = true;
541 
542  iF.erase(it->second.section, it->second.keyword); //Erase it from the iniFile map.
543  }
544  }
545 
546  //here set aside non-deleted iF entries
547  for( auto iFit = iF.names.begin(); iFit != iF.names.end(); ++iFit)
548  {
549  //Insert or update existing
550  m_unusedConfigs[iFit->first].name = iFit->first;
551 
552  std::string sect, nam;
553  iniFile::parseKey(sect, nam, iFit->first);
554  m_unusedConfigs[iFit->first].section = sect;
555  m_unusedConfigs[iFit->first].keyword = nam;
556 
557  m_unusedConfigs[iFit->first].values.push_back(iFit->second);
558 
559  if(m_sources) m_unusedConfigs[iFit->first].sources.push_back(fname);
560 
561  m_unusedConfigs[iFit->first].set = true;
562  }
563 
564  return 0;
565 }

611 int appConfigurator::get( typeT & v,
612  const std::string & name,
613  size_t i,
614  std::unordered_map<std::string, configTarget> & targets
615  )
616 {
617  targets[name].used = true; //This means this was checked.

621  if(configLog)
622  {
623  configLog(name, meta::typeDescription<typeT>::code(), ioutils::convertToString(v), "default");
624  }
625 
626  return 0;
627  }
628 
629  if( targets[name].values.size() <= i) return -1;
630

632

635  {

639  }
640  else
641  {
642  configLog(name, meta::typeDescription<typeT>::code(), targets[name].values[i], "");
643  }
644  }
645 
646  return 0;

692  const std::string & name,
693  size_t i,
694  std::unordered_map<std::string, configTarget> & targets
695  )
696 {
697  targets[name].used = true; //This means this was checked.
698 
699  if(!isSet(name, targets))
700  {
701  if(configLog)
702  {
703  configLog(name, meta::typeDescription<typeT>::code(), "[need a vector to string]", "default");
704  }
705 
706  return 0;
707  }
708 
709  if( targets[name].values.size() <= i) return -1;
710 
711  std::string s;
712 
713 
714  s = ioutils::convertFromString<std::string>(targets[name].values[i]);
715 
716  //if( get<std::string>(s, name, i, targets) < 0) return -1;
717 
718  //Case that s was set to be empty.
719  if(s.size() == 0)
720  {
721  if(configLog)
722  {
723  if(m_sources)
724  {
725  configLog(name, meta::typeDescription<typeT>::code(), "", targets[name].sources[i]);
726  }
727  else
728  {
729  configLog(name, meta::typeDescription<typeT>::code(), "", "");
730  }
731  }
732 
733  v.clear(); //We clear the vector passed as default
734  return 0;
735  }
736 
737  size_t st;
738  size_t com;
739 
740  st = 0;
741 
742  while(::isspace(s[st]) && st < s.size()-1) ++st;
743 
744  com = s.find(',', st);
745 
746  v.clear();
747 
748  while(com != std::string::npos)
749  {
750  v.push_back( ioutils::convertFromString<typeT>(s.substr(st, com-st)) );
751  st = com + 1;
752  while(::isspace(s[st]) && st < s.size()-1) ++st;
753 
754  com = s.find(',', st);
755  }
756  v.push_back( ioutils::convertFromString<typeT>(s.substr(st, s.size()-st)));
757 
758  //Log it here.
759  if(configLog)
760  {
761  if(m_sources)
762  {
763  configLog(name, meta::typeDescription<typeT>::code(), targets[name].values[i], targets[name].sources[i]);
764  }
765  else
766  {
767  configLog(name, meta::typeDescription<typeT>::code(), targets[name].values[i], "");
768  }
769  }
770 
771  return 0;

874 /// A simple config file writing function, useful for testing.
875 /** Write a config file to the path specified by `fname`. Each of the parameters
876  * is a vector, and each component is required of each vector.
877  *
878  * Example:
879  *\code
880  writeConfigFile( "/tmp/test.conf", {"", "", "sect1", "sect1", "sect2", "sect2"},
881  {"key0", "key1", "key2", "key3", "key4", "key5"},
882  {"val0", "val1", "val2", "val3", "val4", "val5"} );
883 
884  *\endcode
885  *results in the file `/tmp/test.conf' containing
886  * \verbatim
887  key0=val0
888  key1=val1
889 
890  [sect1]
891  key2=val2
892  key3=val3
893 
894  [sect2]
895  key4=val4
896  key5=val5
897 
898  * \endverbatim
899  *
900  * No error checking is done.
901  */
902 inline
903 void writeConfigFile( const std::string & fname, ///< [in] the complete path for the output file.
904  const std::vector<std::string> & sections, ///< [in] sections, one per config value
905  const std::vector<std::string> & keywords, ///< [in] keywords, one per config value
906  const std::vector<std::string> & values ///< [in] the values
907  )
908 {
909  std::ofstream fout;
910 
911  fout.open(fname);
912 
913  std::string lastSection;
914  for(size_t i=0; i< sections.size(); ++i)
915  {
916  std::string currSection = sections[i];
917 
918  if( currSection != lastSection && currSection != "")
919  {
920  fout << "\n[" << currSection << "]\n";
921  }
922 
923  fout << keywords[i] << "=" << values[i] << "\n";
924 
925  lastSection = currSection;
926  }
927 
928  fout.close();
929 
930  return;
931 }
