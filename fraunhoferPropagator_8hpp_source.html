41 /// Class to perform Fraunhofer propagation between pupil and focal planes
42 /** This class uses the FFT to propagate between planes, and normalizes so that flux
43  * is conserved. For propagation from pupil to focal plane, the pupil wavefront is tilted so that
44  * the focal-plane image is centered at the geometric center of the array. After propagation from
45  * focal plane to pupil plane, the pupil plane wavefront is un-tilted to restore the
46  * pupil to its original position.
47  *
48  * \tparam _wavefrontT is an Eigen::Array-like type, with std::complex values.
49  *
50  * \todo check if we should reverse the FFT orders to be proper
51  *
52  * \ingroup imaging
53  */

238 template<typename wavefrontT>
239 void fraunhoferPropagator<wavefrontT>::setWavefrontSizePixels(int wfsPix)
240 {
241  //If no change in size, do nothing
242  if(wfsPix == m_centerFocal.rows()) return;
243 
244  m_wavefrontSizePixels = wfsPix;
245 
246  m_xcen = 0.5*(wfsPix - 1.0);
247  m_ycen = 0.5*(wfsPix - 1.0);
248 
249  makeShiftPhase();
250 
251  m_fft_fwd.plan(wfsPix, wfsPix);
252 
253  m_fft_back.plan(wfsPix, wfsPix, MXFFT_BACKWARD);
254 }

256 template<typename wavefrontT>
257 void fraunhoferPropagator<wavefrontT>::makeShiftPhase()
258 {
259  realT pi = boost::math::constants::pi<realT>();
260 
261  //The normalization is included in the tilt.
262  realT norm = 1./(m_wavefrontSizePixels*sqrt(2));
263  complexT cnorm = complexT(norm, norm);
264 
265  //Resize the center phases
266  m_centerFocal.resize(m_wavefrontSizePixels, m_wavefrontSizePixels);
267  m_centerPupil.resize(m_wavefrontSizePixels, m_wavefrontSizePixels);
268 
269  //Shift by 0.5 pixels
270  realT arg = -2.0*pi*0.5*(m_wavefrontSizePixels-m_wholePixel)/(m_wavefrontSizePixels-1);
271 
272  for(int ii=0; ii < m_wavefrontSizePixels; ++ii)
273  {
274  for(int jj=0; jj < m_wavefrontSizePixels; ++jj)
275  {
276  m_centerFocal(ii,jj) = cnorm*exp(complexT(0.,arg*((ii-m_xcen)+(jj-m_ycen))));
277  m_centerPupil(ii,jj) = cnorm*exp(complexT(0., 0.5*pi - arg*((ii-m_xcen)+(jj-m_ycen))));
278  }
279  }
280 }
