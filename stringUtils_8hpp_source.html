51 ///Convert a numerical value to a string
52 /** The default version uses the stream library to convert. A specialization is provided to prevent conversion of std::string.
53  *
54  * The precision parameter is only used for floating point types. If the default precision==0 is passed, then the maximum useful precision is
55  * used, from the value of std::numeric_limits<typeT>::max_digits10.
56  *
57  * The width and pad template parameters can be used to set a fixed maximum width and a pad character.
58  *
59  * Examples:
60  *
61  * To convert a double precision number to string:
62  * \code
63  double d = 2.898434;
64  std::string str;
65  str = convertToString(d); //note that you will not normally need to specify <typeT>
66  * \endcode
67  *
68  * To produce a fixed width 0 padded string from an integer:
69  * \code
70  int i = 23;
71  std::string str;
72  str = convertToString<int, 5, '0'>(i); //result is "00023".
73  * \endcode
74  *
75  * \tparam typeT is the type of value to convert
76  * \tparam width specifies the maximum width, not including the '\0'
77  * \tparam pad specifies the pad character
78  *
79  * \returns a string representation of value
80  *
81  */
82 template< typename typeT, unsigned width=0, char pad=' '>
83 std::string convertToString( const typeT & value, ///< [in] the value of type typeT to be converted
84  int precision = 0 ///< [in] [optional] the precision (see http://www.cplusplus.com/reference/ios/ios_base/precision/) to use for floating point types.
85  )
86 {
87 
88  std::ostringstream convert;
89 
90  if( std::is_floating_point<typeT>::value )
91  {
92  if(precision == 0 )
93  {
94  convert.precision( std::numeric_limits<typeT>::max_digits10);
95  }
96  else
97  {
98  convert.precision(precision);
99  }
100  }
101 
102  convert << value;
103 
104  if(width == 0)
105  {
106  return convert.str();
107  }
108  else
109  {
110  std::string c = convert.str();
111 
112  if( c.length() > width )
113  {
114  c.erase( width, c.length()-width);
115  return c;
116  }
117 
118  return std::string( width-c.length(), pad) + c;
119  }

323 /// Template specialization of convertFromString for bool
324 /** First looks for 0/1,f/t, or F/T in the first non-space character of str.
325  * otherwise uses convertFromString<int>.
326  *
327  * \returns the converted numerical value.
328  */
329 template<> inline
330 bool convertFromString<bool>(const std::string & str /**< [in] is the string to convert*/ )
331 {
332  char c = str[0];
333  size_t i=0;
334  while(isspace(c) && i < str.length()) c = str[i++];
335 
336  if(c == '0' || c == 'f' || c == 'F') return false;
337  if(c == '1' || c == 't' || c == 'T') return true;
338 
339  return (bool) convertFromString<int>(str);
340 }

444 /// Wrap a string by breaking it into smaller sized portions of a desired width
445 /** Whenever possible breaks at spaces. A single space is discarded at the break.
446  */
447 inline
448 int stringWrap( std::vector<std::string> & lines, ///< [out] each new entry contains a wrapped portion of the string. Not cleared, so can accumulate.
449  const std::string & str, ///< [in] the string to wrap
450  int width ///< [in] the maximum width of the output strings
451  )
452 {
453  int L = str.length();
454 
455  if(L == 0) return 0;
456  int startPos, tmpPos, endPos;
457 
458  bool done = false;
459 
460  startPos = 0;
461 
462  while( !done )
463  {
464  if(startPos == L) --startPos; //Just to make sure
465 
466  endPos = startPos + width;
467 
468  if(endPos >= L)
469  {
470  lines.push_back( str.substr( startPos, L-startPos ));
471  done = true;
472  }
473  else
474  {
475  //Backup to nearest space
476  tmpPos = endPos;
477  while( !isspace(str[tmpPos]) && tmpPos >= startPos ) --tmpPos;
478 
479  //If we aren't at the beginning (i.e. splitting consecutive characters) we use new end position
480  if(tmpPos > startPos) endPos = tmpPos;
481 
482 
483  lines.push_back( str.substr( startPos, endPos-startPos) );
484 
485 
486  startPos = endPos;
487 
488  //Clear 1 space
489  if( str[startPos] == ' ') ++startPos;
490  }
491  }
492 
493  return 0;
494 }

496 /// Parses a string into a vector of tokens delimited by a character
497 /** E.g., the string
498  * \code
499  std::string s={"0,1,2,3,4"};
500  std::vector<int> v;
501  parseStringVector(v,s);
502  \endcode
503  * is parsed to a vector as if it was initialized with
504  \code
505  std::vector<int> v = {0,1,2,3,4};
506  \endcode
507  *
508  * \tparam typeT the type to convert the tokens too.
509  */
510 template<typename typeT>
511 void parseStringVector( std::vector<typeT> & v, ///< [out] the vector holding the parsed and converted tokens. Is cleared.
512  const std::string & s, ///< [in] the string to parse
513  char delim = ',' ///< [in] [optional] the delimiter. Default is comma \p ','.
514  )
515 {
516  size_t st;
517  size_t com;
518 
519  st = 0;
520  com = s.find(delim, st);
521 
522  v.clear();
523 
524  while(com != std::string::npos)
525  {
526  v.push_back( convertFromString<typeT>(s.substr(st, com-st)) );
527  st = com + 1;
528  com = s.find(delim, st);
529  }
530  v.push_back( convertFromString<typeT>(s.substr(st, s.size()-st)));
531 
532 }

544  \endcode
545  *

547  */

567  }
