42 /** \defgroup moffat_peak_fit Moffat Functions
43  * \brief Fitting Moffat functions to data.
44  *
45  * The Moffat Function\cite moffat_1969, a.k.a. the Moffat Profile, a.k.a. the Moffat Distribution, has the form
46  * \f[
47  I(x) = I_{pk}\left[ 1 + \frac{x^2}{\alpha^2}\right]^{-\beta}
48  * \f]
49  * With \f$\beta=1\f$ it is the
50  * Lorentzian or Cauchy distribution. See also https://en.wikipedia.org/wiki/Moffat_distribution and
51  * https://en.wikipedia.org/wiki/Cauchy_distribution.
52  *
53  * For utilities for working with Moffat functions see \ref gen_math_moffats.
54  *
55  * \ingroup peak_fit
56  */

444 ///\ref levmarInterface fitter structure for the symmetric Moffat.
445 /** \ingroup moffat_peak_fit
446  *
447  */
448 template<typename _realT>
449 struct moffat2D_sym_fitter
450 {
451  typedef _realT realT;
452 
453  static const int nparams = 6;
454 
455  static void func(realT *p, realT *hx, int m, int n, void *adata)
456  {
457  array2FitMoffat<realT> * arr = (array2FitMoffat<realT> *) adata;
458 
459  size_t idx_mat, idx_dat;
460 
461  idx_dat = 0;
462 
463  realT I0 = arr->I0(p);
464  realT I = arr->I(p);
465  realT x0 = arr->x0(p);
466  realT y0 = arr->y0(p);
467  realT alpha = arr->alpha(p);
468  realT beta = arr->beta(p);
469 
470  for(int i=0; i<arr->nx; ++i)
471  {
472  for(int j=0; j<arr->ny; ++j)
473  {
474  idx_mat = i+j*arr->nx;
475 
476  hx[idx_dat] = func::moffat2D<realT>(i,j,I0,I, x0, y0, alpha, beta) - arr->data[idx_mat];
477 
478  ++idx_dat;
479  }
480  }
481  }
482 };
