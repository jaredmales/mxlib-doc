36 /** Calculates the frequency in bins with uniform spacing.
37  *
38  * \tparam realT the real data type
39  *
40  * \ingroup gen_math
41  */
42 template<typename realT>
43 class histogramUniform
44 {
45 public:
46 
47  realT _min; ///<The mininum bin location
48  realT _max; ///<The maximum bin location
49  realT _width; ///<The bin width
50 
51  std::vector<realT> _freqs; ///<The frequencies, one for each bin.
52 
53  ///Setup the histogram, performing allocations.
54  void setup( realT mn, ///< [in] the new minimum bin location
55  realT mx, ///< [in] the new maximum bin location
56  realT w ///< [in] the bin width
57  )
58  {
59  _min = mn;
60  _max = mx;
61  _width = w;
62 
63  reset();
64  }
65 
66  ///Resize and 0 the frequency vector. Assumes _min, _max, and _width are set.
67  void reset()
68  {
69  _freqs.resize( (_max - _min)/_width + 1, 0);
70  }
71 
72  ///Accumulate a value in the appropriate bin.
73  void accum( const realT & val /**< [in] The value to accumulate */)
74  {
75  int i = (val - _min) / _width;
76  if( i < 0) i = 0;
77  if( i >= _freqs.size()) i = _freqs.size()-1;
78 
79 
80  ++_freqs[i];
81  }
82 
83  ///Accumulate a vector of values.
84  void accum( const std::vector<realT> & vals /**< [in] The vector of values to accumulate */)
85  {
86  for(int i=0; i< vals.size(); ++i) accum(vals[i]);
87  }
88 
89  ///Get the frequency in the i-th bin.
90  realT freq(int i /**< [in] the bin number */)
91  {
92  return _freqs[i]; ///\returns the current value of _freqs[i].
93  }
94 
95  ///Get the number of bins
96  int bins()
97  {
98  return _freqs.size(); ///\returns the size of the frequency vector.
99  }
100 
101  ///Get the value of the left-edge of the i-th bin.
102  realT binLeft(int i /**< [in] the bin number */)
103  {
104  return _min + i*_width; ///\returns the value of the left-edge of the i-th bin.
105  }
106 
107  ///Get the value of the middle of the i-th bin.
108  realT binMid(int i /**< [in] the bin number */)
109  {
110  return _min + i*_width + 0.5*_width; ///\returns the value of the middle of the i-th bin.
111  }
112 
113  ///Get the value of the right edge of the i-th bin.
114  realT binRight(int i /**< [in] the bin number */)
115  {
116  return _min + i*_width + 1.0*_width; ///\returns the value of the right edge of the i-th bin.
117  }
118 
119  ///Normalize the current frequencies so that the integral over all bins is 1.
120  /** This normalizes the histogram so that it is a probability distribution, such that the sum
121  * \f$ \sum_i P_i \Delta x = 1 \f$ where \f$ \Delta x \f$ is the bin width.
122  */
123  void normalize( int excludeTop = 0 /**< [in] [optional] specifies a number of bins at the top of the range to exclude from the sum */)
124  {
125  realT sum = 0;
126 
127  for(int i=0; i< _freqs.size()-excludeTop; ++i)
128  {
129  sum += _freqs[i];
130  }
131 
132  for(int i=0; i< _freqs.size(); ++i)
133  {
134  _freqs[i] /= (sum*_width);
135  }
136  }
137 
138 };
