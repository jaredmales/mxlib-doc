167 int fitsHeaderCard::write(fitsfile *fptr)
168 {
169  switch(type)
170  {
171  case TSTRING:
172  {
173  return fits_write_key<char *>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
174  }
175  case TBYTE:
176  {
177  unsigned char b = Int();
178  return fits_write_key<unsigned char>(fptr, (char *) keyword.c_str(), &b, (char *) comment.c_str());
179  }
180  case TSBYTE:
181  {
182  signed char sc = Int();
183  return fits_write_key<signed char>(fptr, (char *) keyword.c_str(), &sc, (char *) comment.c_str());
184  }
185  case TSHORT:
186  {
187  short s = Int();
188  return fits_write_key<short>(fptr, (char *) keyword.c_str(), &s, (char *) comment.c_str());
189  }
190  case TUSHORT:
191  {
192  unsigned short us = Int();
193  return fits_write_key<unsigned short>(fptr, (char *) keyword.c_str(), &us, (char *) comment.c_str());
194  }
195  case TINT:
196  {
197  int i = Int();
198  return fits_write_key<int>(fptr, (char *) keyword.c_str(), &i, (char *) comment.c_str());
199  }
200  case TLONG:
201  {
202  long l = Long();
203  return fits_write_key<long>(fptr, (char *) keyword.c_str(), &l, (char *) comment.c_str());
204  }
205  case TLONGLONG:
206  {
207  long long ll = longLong();
208  return fits_write_key<long long>(fptr, (char *) keyword.c_str(), &ll, (char *) comment.c_str());
209  }
210  case TULONG:
211  {
212  unsigned long ul = unsignedLong();
213  return fits_write_key<unsigned long>(fptr, (char *) keyword.c_str(), &ul, (char *) comment.c_str());
214  }
215  case TFLOAT:
216  {
217  float f = Float();
218  return fits_write_key<float>(fptr, (char *) keyword.c_str(), &f, (char *) comment.c_str());
219  }
220  case TDOUBLE:
221  {
222  double d = Double();
223  return fits_write_key<double>(fptr, (char *) keyword.c_str(), &d, (char *) comment.c_str());
224  }
225  case fitsTCOMMENT:
226  {
227  return fits_write_comment(fptr, (char *) comment.c_str());
228  }
229  case fitsTHISTORY:
230  {
231  return fits_write_history(fptr, (char *) comment.c_str());
232  }
233  default:
234  {
235  return fits_write_key<fitsUnknownType>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
236  }
237  }
238 }
