42/// Dump the git status of a repository to a stream
43/** Prints the provided SHA1 hash and whether or not the library
44 * has been modified since that commit.
45 *
46 * \tparam iosT a std::ostream-like type
47 * \tparam comment a character to print at the beginning of each line. Default is '#'.
48 */
49template <typename iosT, char comment = '#'>
50iosT &dumpGitStatus(
51 iosT &ios, ///< [in] a std::ostream-like stream.
52 const std::string &repoName, ///< [in] The name of the repository
53 const std::string &branch, ///< [in] The name of the branch
54 const std::string &sha1, ///< [in] The sha1 hash for the current commit
55 const bool &modified, ///< [in] Whether or not the repository is currently modified
56 const std::string &section = "" ///< [in] [optional] Descriptive sub-section name (e.g. headers vs. compiled)
57)
58{
59 // This causes the stream to not output a '\0' if comment = '\0'
60 char c[] = { comment, '\0' };
61
62 bool sect = !( section == "" );
63 std::string space = " ";
64 if( sect )
65 space += " ";
66
67 ios << c << "--------------------------------------------\n";
68 ios << c << " " << repoName << " git status:\n";
69 if( sect )
70 ios << c << " " << section << ":\n";
71 ios << c << space << "branch: " << branch << "\n";
72 ios << c << space << "SHA1: " << sha1 << "\n";
73 ios << c << space << "modified flag: " << std::boolalpha << modified << "\n";
74 ios << c << "--------------------------------------------\n";
75
76 return ios;
77}

84 * \tparam comment a character to print at the beginning of each line. Default is '#'.
