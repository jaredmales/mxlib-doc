76 ///Transformation by cubic convolution interpolation
77 /** Uses the cubic convolution interpolation kernel. See <a href="https://en.wikipedia.org/wiki/Bicubic_interpolation">https://en.wikipedia.org/wiki/Bicubic_interpolation </a>.
78  *
79  * The parameter \ref cubic should be left as the default -0.5 in most cases, which gives the bicubic spline interpolator. See
80  * <a href="https://en.wikipedia.org/wiki/Cubic_Hermite_spline">https://en.wikipedia.org/wiki/Cubic_Hermite_spline</a>.
81  *
82  * \tparam _arithT is the type in which to do all calculations. Should be a floating point type.
83  *
84  * \ingroup image_transforms
85  */
86 template<typename _arithT>
87 struct cubicConvolTransform
88 {
89  typedef _arithT arithT; ///< The type in which all calculations are performed.
90 
91  static const size_t width = 4;
92  static const size_t lbuff = 1;
93 
94  arithT cubic {-0.5}; ///< The kernel parameter. The default value -0.5 gives the bicubic spline interpolator.
95 
96  explicit cubicConvolTransform(arithT c)
97  {
98  cubic = c;
99  }
100 
101  cubicConvolTransform() {}
102 
103  cubicConvolTransform(const cubicConvolTransform & t)
104  {
105  cubic = t.cubic;
106  }
107 
108  arithT cubicConvolKernel(arithT d)
109  {
110  if(d <= 1) return (cubic+2.)*d*d*d - (cubic+3.)*d*d + 1.;
111 
112  if(d < 2) return cubic*d*d*d -5.*cubic*d*d + 8.*cubic*d - 4.*cubic;
113 
114  return 0;
115  }
116 
117  template<typename arrT, typename arithT>
118  void operator()(arrT & kern, arithT x, arithT y)
119  {
120  arithT km2x,km1x,kp1x,kp2x;
121  arithT km2y,km1y,kp1y,kp2y;
122 
123  km2x = cubicConvolKernel((1.+x));
124  km1x = cubicConvolKernel(x);
125  kp1x = cubicConvolKernel(1.-x);
126  kp2x = cubicConvolKernel(2.-x);
127 
128  km2y = cubicConvolKernel((1.+y));
129  km1y = cubicConvolKernel(y);
130  kp1y = cubicConvolKernel(1.-y);
131  kp2y = cubicConvolKernel(2.-y);
132 
133  kern(0,0) = km2x*km2y;
134  kern(0,1) = km2x*km1y;
135  kern(0,2) = km2x*kp1y;
136  kern(0,3) = km2x*kp2y;
137 
138  kern(1,0) = km1x*km2y;
139  kern(1,1) = km1x*km1y;
140  kern(1,2) = km1x*kp1y;
141  kern(1,3) = km1x*kp2y;
142 
143  kern(2,0) = kp1x*km2y;
144  kern(2,1) = kp1x*km1y;
145  kern(2,2) = kp1x*kp1y;
146  kern(2,3) = kp1x*kp2y;
147 
148  kern(3,0) = kp2x*km2y;
149  kern(3,1) = kp2x*km1y;
150  kern(3,2) = kp2x*kp1y;
151  kern(3,3) = kp2x*kp2y;
152  }
153 };

173 /// Rotate an image represented as an eigen array
174 /** Uses the given transformation type to rotate an image.
175  *
176  * \tparam transformT specifies the transformation to use [will be resolved by compiler]
177  * \tparam arrT is the eigen array type of the output [will be resolved by compiler]
178  * \tparam arrT2 is the eigen array type of the input [will be resolved by compiler]
179  * \tparam floatT is a floating point type [will be resolved by compiler in most cases]
180  *
181  */
182 template<typename transformT, typename arrT, typename arrT2, typename floatT>
183 void imageRotate( arrT & transim, ///< [out] The rotated image. Must be pre-allocated.
184  const arrT2 &im, ///< [in] The image to be rotated.
185  floatT dq, ///< [in] the angle, in radians, by which to rotate in the c.c.w. direction
186  transformT trans ///< [in] is the transformation to use
187  )
188 {
189  typedef typename transformT::arithT arithT;
190  arithT cosq, sinq;
191  arithT x0, y0, x,y;
192  arithT xcen, ycen;
193 
194  int Nrows, Ncols;
195 
196  int i0, j0;
197 
198  const int lbuff = transformT::lbuff;
199  const int width = transformT::width;
200 
201  cosq = cos(dq);
202  sinq = sin(dq);
203 
204  Nrows = im.rows();
205  Ncols = im.cols();
206 
207 
208  transim.resize(Nrows, Ncols);
209 
210  //The geometric image center
211  xcen = 0.5*(Nrows-1.);
212  ycen = 0.5*(Ncols-1.);
213 
214  int xulim = Nrows-width+lbuff;// - 1;
215  int yulim = Ncols-width+lbuff;// - 1;
216 
217  arithT xc_x_cosq = xcen*cosq;
218  arithT xc_x_sinq = xcen*sinq;
219  arithT yc_x_cosq = ycen*cosq;
220  arithT yc_x_sinq = ycen*sinq;
221 
222  xc_x_cosq += yc_x_sinq;
223  xc_x_sinq -= yc_x_cosq;
224 
225 
226  #ifdef MXLIB_USE_OMP
227  #pragma omp parallel private(x0,y0,i0,j0,x,y)
228  #endif
229  {
230  arithT i_x_cosq, i_x_sinq;
231  arrT kern;
232  kern.resize(width,width);
233 
234  #ifdef MXLIB_USE_OMP
235  #pragma omp for schedule(static, 1)
236  #endif
237  for(int i=0;i<Nrows; ++i)
238  {
239  i_x_cosq = i*cosq - xc_x_cosq;// + xcen;
240  i_x_sinq = -(i*sinq - xc_x_sinq);// + ycen;
241 
242  for(int j=0;j<Ncols; ++j)
243  {
244  //We are actually doing this rotation matrix:
245  //x0 = (i-xcen)*cosq + (j-ycen)*sinq;
246  //y0 = -(i-xcen)*sinq + (j-ycen)*cosq;
247  //This is the minimum-op representation of the above rotation matrix:
248  x0 = i_x_cosq + j*sinq;
249  y0 = i_x_sinq + j*cosq;
250 
251  //Get lower left index
252  i0 = x0 +xcen;
253  j0 = y0 +ycen;
254 
255  if(i0 <= lbuff || i0 >= xulim || j0 <= lbuff || j0 >= yulim)
256  {
257  transim(i,j) = 0;
258  continue;
259  }
260 
261  //Get the residual
262  x = x0+xcen-i0;
263  y = y0+ycen-j0;
264 
265  trans(kern, x, y);
266  transim(i,j) = (im.block(i0-lbuff,j0-lbuff, width, width) * kern).sum();
267  }//for j
268  }//for i
269  }//#pragma omp parallel

273 /// Shift an image by whole pixels, wrapping around..
274 /** The output image can be smaller than the input image, in which case the wrapping still occurs for the input image, but only
275  * output images worth of pixels are actually shifted. This is useful, for instance, when propagating large turbulence phase screens
276  * where one only needs a small section at a time.
277  *
278  * \tparam outputArrT is the eigen array type of the output [will be resolved by compiler]
279  * \tparam inputArrT is the eigen array type of the input [will be resolved by compiler]
280  *
281  */
282 template<typename outputArrT, typename inputArrT>
283 void imageShiftWP( outputArrT & out, ///< [out] contains the shifted image. Must be pre-allocated, but can be smaller than the in array.
284  inputArrT & in, ///< [in] the image to be shifted.
285  int dx, ///< [in] the amount to shift in the x direction
286  int dy ///< [in] the amount to shift in the y direction
287  )
288 {
289  dx %= in.rows();
290  dy %= in.cols();
291 
292  int outr = out.rows();
293  int outc = out.cols();
294  int inr = in.rows();
295  int inc = in.cols();
296  #ifdef MXLIB_USE_OMP
297  #pragma omp parallel
298  #endif
299  {
300  int x, y;
301 
302  #ifdef MXLIB_USE_OMP
303  #pragma omp for
304  #endif
305  for(int cc=0;cc<outc; ++cc)
306  {
307  y = cc - dy;
308 
309  if (y < 0) y += inc;
310  else if (y >= inc) y -= inc;
311 
312  for(int rr=0; rr<outr; ++rr)
313  {
314  x = rr - dx;
315 
316  if(x < 0) x += inr;
317  else if (x >= inr) x -= inr;
318 
319  out(rr, cc) = in(x,y);
320  }
321  }

323  }

383 /// Shift an image.
384 /** Uses the given transformation type to shift an image.
385  *
386  * Note that this does not treat the edges
387  * of the image, determined by the buffer width (lbuff) of the kernel and the size of shift. If you wish to
388  * treat the edges, you must pad the image by at least lbuff+abs(shift) pixels in each direction, and
389  * implement a strategy (zeros, mirror, wrap) prior to calling this function.
390  *
391  * \tparam arrOutT is the Eigen-like array type of the output [will be resolved by compiler]
392  * \tparam arrInT is the Eigen-like array type of the input [will be resolved by compiler]
393  * \tparam floatT1 is a floating point type [will be resolved by compiler]
394  * \tparam floatT2 is a floating point type [will be resolved by compiler]
395  * \tparam transformT specifies the transformation to use [will be resolved by compiler]
396  *
397  */
398 template<typename arrOutT, typename arrInT, typename floatT1, typename floatT2, typename transformT>
399 void imageShift( arrOutT & transim, ///< [out] Will contain the shifted image. Will be allocated.
400  const arrInT &im, ///< [in] the image to be shifted.
401  floatT1 dx, ///< [in] the amount to shift in the x direction
402  floatT2 dy, ///< [in] the amount to shift in the y direction
403  transformT trans ///< [in] trans is the transformation to use
404  )
405 {
406  typedef typename transformT::arithT arithT;
407 
408  int Nrows, Ncols;
409 
410  const int lbuff = transformT::lbuff;
411  const int width = transformT::width;
412 
413  Nrows = im.rows();
414  Ncols = im.cols();
415 
416  int xulim = Nrows-width+lbuff;
417  int yulim = Ncols-width+lbuff;
418 
419  transim.resize(Nrows, Ncols);
420 
421  #ifdef MXLIB_USE_OMP
422  #pragma omp parallel
423  #endif
424  {
425  int i0, j0;
426  // (rx, ry) is fractional residual of shift
427  arithT rx = 1-(dx-floor(dx));
428  arithT ry = 1-(dy-floor(dy));
429 
430  arrOutT kern;
431  kern.resize(width,width);
432  trans(kern, rx, ry);
433 
434  #ifdef MXLIB_USE_OMP
435  #pragma omp for
436  #endif
437  for(int i=0;i<Nrows; ++i)
438  {
439  // (i,j) is position in new image
440  // (i0,j0) is integer position in old image
441 
442  i0 = i-dx;
443 
444  if(i0 <= lbuff || i0 >= xulim)
445  {
446  for(int j=0;j<Ncols; ++j)
447  {
448  transim(i,j) = 0;
449  }
450  continue;
451  }
452 
453  for(int j=0;j<Ncols; ++j)
454  {
455  j0 = j-dy;
456 
457  if(j0 <= lbuff || j0 >= yulim)
458  {
459  transim(i,j) = 0;
460  continue;
461  }
462 
463  transim(i,j) = (im.block(i0-lbuff,j0-lbuff, width, width) * kern).sum();
464  }//for j
465  }//for i
466  }//#pragam omp

471 /// Magnify an image.
472 /** Uses the given transformation type to magnify the input image to the size of the output image.
473  *
474  * Here we assume that the image center is the mxlib standard:
475  * \code
476  x_center = 0.5*(im.rows()-1);
477  y_center = 0.5*(im.cols()-1);
478  * \endcode
479  * Some care is necessary to prevent magnification from shifting the image with respect to this center. The main result is that the
480  * magnification factors (which can be different in x and y) are defined thus:
481  * \code
482  x_mag = (transim.rows()-1.0) / (im.rows()-1.0);
483  y_mag = (transim.cols()-1.0) / (im.cols()-1.0);
484  * \endcode
485  *
486  * Example:
487  * \code
488  im1.resize(512,512);
489  //add image to im1
490  im2.resize(1024,1024);
491  imageMagnify(im2,im1, cubicConvolTransform<double>());
492  \endcode
493  * In this exmple, the image in im1 will be magnified by `1023.0/511.0 = 2.002x` and placed in im2.
494  *
495  * This transform function does not handle edges. If treatment of edges is desired, you must pad the input
496  * image using the desired strategy before calling this function. Note that the padded-size of the input image
497  * will affect the magnification factor.
498  *
499  * \tparam arrOutT is the eigen array type of the output.
500  * \tparam arrInT is the eigen array type of the input.
501  * \tparam transformT specifies the transformation to use.
502  */
503 template<typename arrOutT, typename arrInT, typename transformT>
504 void imageMagnify( arrOutT & transim, ///< [out] contains the magnified image. Must be pre-allocated.
505  const arrInT &im, ///< [in] is the image to be magnified.
506  transformT trans ///< [in] is the transformation to use
507  )
508 {
509  typedef typename transformT::arithT arithT;
510 
511  arithT x0, y0, x,y;
512 
513  int Nrows, Ncols;
514 
515  int i0, j0;
516 
517  const int lbuff = transformT::lbuff;
518  const int width = transformT::width;
519 
520  Nrows = transim.rows();
521  Ncols = transim.cols();
522 
523  int xulim = im.rows()-lbuff - 1;
524  int yulim = im.cols()-lbuff - 1;
525 
526  arithT x_scale = ( (arithT) im.rows()-1.0)/ (transim.rows()-1.0); //this is 1/x_mag
527  arithT y_scale = ( (arithT) im.cols()-1.0)/ (transim.cols()-1.0); //this is 1/y_mag
528 
529  arithT xcen = 0.5*( (arithT) transim.rows() - 1.0);
530  arithT ycen = 0.5*( (arithT) transim.cols() - 1.0);
531 
532  arithT xcen0 = 0.5*( (arithT) im.rows() - 1.0);
533  arithT ycen0 = 0.5*( (arithT) im.cols() - 1.0);
534 
535 // #pragma omp parallel private(x0,y0,i0,j0,x,y) num_threads(4)
536  {
537  arrOutT kern;
538  kern.resize(width,width);
539 
540  for(int j=0;j<Ncols; ++j)
541  {
542  // (i,j) is position in new image
543  // (x0,y0) is true position in old image
544  // (i0,j0) is integer position in old image
545  // (x, y) is fractional residual of (x0-i0, y0-j0)
546 
547  y0 = ycen0 + (j-ycen)*y_scale;
548  j0 = y0;
549 
550  if(j0 < lbuff || j0 >= yulim)
551  {
552  for(int i=0;i<Nrows; ++i)
553  {
554  transim(i,j) = 0;
555  }
556  continue;
557  }
558 
559 // #pragma omp for
560  for(int i=0;i<Nrows; ++i)
561  {
562  x0 = xcen0 + (i-xcen)*x_scale;
563  i0 = x0; //just converting to int
564 
565  if(i0 < lbuff || i0 >= xulim)
566  {
567  transim(i,j) = 0;
568  continue;
569  }
570 
571 
572  //Get the residual
573  x = x0-i0;
574  y = y0-j0;
575 
576  trans(kern, x, y);
577  transim(i,j) = (im.block(i0-lbuff,j0-lbuff, width, width) * kern).sum();
578  }//for j
579  }//for i
580  }//#pragma omp
581 
582 }

602 /// Re-bin an image using the sum, reducing its size while conserving the total flux.
603 /** Optionally this can be the mean instead of the sum filter, in which case total flux is not conserved.
604  */
605 template<typename imageOutT, typename imageInT>
606 int imageRebinSum( imageOutT & imout, ///< [out] the re-binned image. Must be allocated to size which is an integer factor smaller than imin.
607  const imageInT & imin, ///< [in] the image to rebin
608  bool mean = false ///< [in] if true the output is the mean rather than the sum.
609  )
610 {

612  if(imin.cols()/imout.cols() != rebin) return -1;
613 
614  int N = 1;
615  if(mean) N = rebin*rebin;
616  for(int i=0;i<imout.rows(); ++i)
617  {
618  for(int j=0; j<imout.cols(); ++j)
619  {
620  imout(i,j) = imin.block( i*rebin, j*rebin, rebin, rebin).sum()/N;
621  }
622  }
623

627 /// Re-bin an image using the sum, reducing its size while conserving the total flux. Records the value and position of the re-binned max pixel.
628 /** Optionally this can be the mean instead of the sum filter, in which case total flux is not conserved.
629  *
630  * \overload

632 template<typename imageOutT, typename imageInT>
633 int imageRebinSum( imageOutT & imout, ///< [out] the re-binned image. Must be allocated to size which is an integer factor smaller than imin.
634  int & xMax, ///< [out] the x-locatioin of the max pixel
635  int & yMax, ///< [out] the y-locatioin of the max pixel
636  typename imageOutT::Scalar & pMax, ///< [out] the value of the max pixel

787 /// Down-sample an image, reducing its size while conserving the total flux.
788 /** If the old size is an integer multiple of the new size, this is just a re-bin. If not an integer multiple,
789  * the image is interpolated after performing the closest re-bin, and then re-normalized to conserve flux.
790  *
791  * \todo Allow selection of interpolator, providing a default version.
792  */
793 template<typename imageOutT, typename imageInT>
794 void imageDownSample(imageOutT & imout, const imageInT & imin)
795 {
796  typedef typename imageOutT::Scalar Scalar;
797 
798 
799  //Record this for normalization later
800  Scalar inputTotal = fabs(imin.sum());
801 
802 
803  //As a first step, rebin to nearest whole pixel factor which is larger than the desired output size
804  int closestRebin = imin.rows()/imout.rows();//, imin.cols()/imout.cols() );
805 
806  float sample = ( (float) imin.rows())/ closestRebin;
807 
808  while( sample != floor(sample))
809  {
810  --closestRebin;
811  if(closestRebin == 1) break;
812  sample = ( (float) imin.rows())/ closestRebin;
813  }
814 
815  //Eigen::Array<Scalar, Eigen::Dynamic, Eigen::Dynamic> temp;
816  imageOutT temp;
817  temp.resize( imin.rows()/closestRebin, imin.cols()/closestRebin);
818 
819  for(int i=0;i<temp.rows(); ++i)
820  {
821  for(int j=0; j<temp.cols(); ++j)
822  {
823  temp(i,j) = imin.block( i*closestRebin, j*closestRebin, closestRebin, closestRebin).sum();
824  }
825  }
826 
827  //If the output image is now the requested size return.
828  if(temp.rows() == imout.rows() && temp.cols() == imout.cols())
829  {
830  imout = temp;
831  return;
832  }
833  //Otherwise, re-sample using bilinear interpolation.
834  typedef bilinearTransform<Scalar> transformT;
835 
836  transformT trans;
837  //Eigen::Array<Scalar, -1,-1> kern;
838  imageOutT kern;
839 
840  const int lbuff = transformT::lbuff;
841  const int width = transformT::width;
842 
843  for(int i=0;i<imout.rows(); ++i)
844  {
845  for(int j=0;j<imout.cols(); ++j)
846  {
847  double x = ( (double) i/ imout.rows())*temp.rows();
848  double y = ( (double) j/ imout.cols())*temp.cols();
849 
850  trans(kern, x-floor(x), y-floor(y));
851 
852  imout(i,j) = (temp.block( floor(x)-lbuff, floor(y)-lbuff, width, width)*kern).sum();
853 
854  }
855  }
856 
857  //Normalize
858  Scalar outputTotal = fabs(imout.sum());
859  imout *= inputTotal/outputTotal;
860 }
