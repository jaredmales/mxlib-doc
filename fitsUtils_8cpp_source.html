34 int fitsStripApost(std::string & s)
35 {
36  int stripped = 0;
37  int p = s.find_first_not_of(" \t\n\t");
38  if(s[p] == '\'')
39  {
40  s.erase(0,p+1);
41  ++stripped;
42  }
43 
44  p = s.find_last_not_of(" \t\n\t");
45  if(s[p] == '\'')
46  {
47  s.erase(p);
48  ++stripped;
49  }
50 
51  --p;
52 
53  while(s[p] == ' ' && p >=0)
54  {
55  s.erase(p);
56  --p;
57  ++stripped;
58  }
59 
60  return stripped;
61 }
