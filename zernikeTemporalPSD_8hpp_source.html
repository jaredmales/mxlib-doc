400 template<typename realT, typename aosysT>
401 int zernikeTemporalPSD<realT, aosysT>::m_multiLayerPSD( std::vector<realT> & PSD,
402  std::vector<realT> & freq,
403  int zern_j,
404  realT fmax,
405  isParallel<true> parallel )
406 {
407  static_cast<void>(parallel);
408 
409  #pragma omp parallel
410  {
411  //Records each layer PSD
412  std::vector<realT> single_PSD(freq.size());
413 
414  #pragma omp for
415  for(size_t i=0; i< m_aosys->atm.n_layers(); ++i)
416  {
417  singleLayerPSD(single_PSD, freq, zern_j, i, fmax);
418 
419  //Now add the single layer PSD to the overall PSD, weighted by Cn2
420  #pragma omp critical
421  for(size_t j=0; j<freq.size(); ++j)
422  {
423  PSD[j] += m_aosys->atm.layer_Cn2(i)*single_PSD[j];
424  }
425  }
426  }
427 
428  return 0;
429 }

482 ///Worker function for GSL Integration for the basic sin/cos Fourier modes.
483 /** \ingroup mxAOAnalytic
484  */
485 template<typename realT, typename aosysT>
486 realT F_zernike (realT kv, void * params)
487 {
488  zernikeTemporalPSD<realT, aosysT> * Fp = (zernikeTemporalPSD<realT, aosysT> *) params;
489 
490  realT f = Fp->m_f;
491  realT v_wind = Fp->m_aosys->atm.layer_v_wind(Fp->_layer_i);
492  realT D = Fp->m_aosys->D();
493 
494  int m, n;
495  sigproc::noll_nm(n, m, Fp->m_zern_j);
496 
497  realT ku = f/v_wind;
498 
499  realT phi = atan(kv/ku);
500 
501  realT k = sqrt( pow(ku,2) + pow(kv,2) );
502 
503  realT Q2norm = sigproc::zernikeQNorm(k*D, phi, n, m);
504 
505  realT P = Fp->m_aosys->psd(Fp->m_aosys->atm, Fp->_layer_i, k, Fp->m_aosys->lam_sci(), Fp->m_aosys->lam_wfs(), Fp->m_aosys->secZeta() );
506 
507 
508  return P*Q2norm ;
509 }
