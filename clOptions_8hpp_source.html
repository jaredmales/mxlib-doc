172 void clOptions::add( const std::string & optName,
173  const char * const shortOpt,
174  const char * const longOpt,
175  int argT
176  )
177 {
178  mapIterator it;
179  it = map.find(optName);
180 
181 
182  if(it == map.end())
183  {
184  if(argT == argType::Optional)
185  {
186  descriptions.push_back({nOpts, argT, shortOpt, longOpt, option::Arg::Optional, ""});
187  }
188  else if(argT == argType::Required)
189  {
190  descriptions.push_back({nOpts, argT, shortOpt, longOpt, Arg_Required, ""});
191  }
192  else
193  {
194  descriptions.push_back({nOpts, argT, shortOpt, longOpt, option::Arg::None, ""});
195  }
196 
197  map.insert({optName, nOpts});
198  typeMap.insert({optName, argT});
199 
200  ++nOpts;
201  return;
202  }

205 inline
206 void clOptions::parse( int argc,
207  char **argv,
208  std::vector<std::string> * nonOptions
209  )
210 {
211  argc-=(argc>0); argv+=(argc>0); // skip program name argv[0] if present
212 
213  //If not already done, we push the unknown catcher and the termination descriptor.
214  if(descriptions.back().index != 0)
215  {
216  descriptions.push_back({nOpts, 0, "", "", option::Arg::None, ""}); //This is inserted to catch unknown options
217  descriptions.push_back({0,0,0,0,0,0});
218  }
219 
220  //Now allocate.
221  option::Stats stats(descriptions.data(), argc, argv);
222  options = new option::Option[stats.options_max];
223  buffer = new option::Option[stats.buffer_max];
224 
225  option::Parser parse(false, descriptions.data(), argc, argv, options, buffer);
226 
227  if(nonOptions)
228  {
229  nonOptions->resize(parse.nonOptionsCount());
230 
231  for(int i=0;i<parse.nonOptionsCount(); ++i)
232  {
233  (*nonOptions)[i] = parse.nonOption(i);
234  }
235  }
236 }

239 const char * clOptions::operator[]( const std::string & key )
240 {
241  mapIterator it = map.find(key);
242 
243  if(it == map.end())
244  {
245  std::cerr << "oh no\n";
246  return 0;
247  }
248 
249  int typeDesc = typeMap[key];
250 
251  //If this option is not pressent, either return the opposite T/F condition or blank
252  if(options[it->second].type() == argType::None)
253  {
254  if(typeDesc == argType::False) return trueStr;
255  if(typeDesc == argType::True) return falseStr;
256  return blankStr;
257  }
258 
259  if(typeDesc == argType::False || typeDesc == argType::True)
260  {
261  if(options[it->second].last()->type() == argType::False) return falseStr;
262  else return trueStr;
263  }
264 
265  if(options[it->second].arg == 0) return blankStr;
266 
267  return options[it->second].last()->arg;
268 }
